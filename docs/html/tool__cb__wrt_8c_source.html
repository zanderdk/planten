34 /* create a local file for writing, return TRUE on success */
35 bool tool_create_output_file(struct OutStruct *outs,
36  bool append)
37 {
38  struct GlobalConfig *global = outs->config->global;
39  FILE *file;
40 
41  if(!outs->filename || !*outs->filename) {
42  warnf(global, "Remote filename has no length!\n");
43  return FALSE;
44  }
45 
46  if(outs->is_cd_filename && !append) {
47  /* don't overwrite existing files */
48  file = fopen(outs->filename, "rb");
49  if(file) {
50  fclose(file);
51  warnf(global, "Refusing to overwrite %s: %s\n", outs->filename,
52  strerror(EEXIST));
53  return FALSE;
54  }
55  }
56 
57  /* open file for writing */
58  file = fopen(outs->filename, append?"ab":"wb");
59  if(!file) {
60  warnf(global, "Failed to create the file %s: %s\n", outs->filename,
61  strerror(errno));
62  return FALSE;
63  }
64  outs->s_isreg = TRUE;
65  outs->fopened = TRUE;
66  outs->stream = file;
67  outs->bytes = 0;
68  outs->init = 0;
69  return TRUE;
70 }

84  /*
85  * Once that libcurl has called back tool_write_cb() the returned value
86  * is checked against the amount that was intended to be written, if
87  * it does not match then it fails with CURLE_WRITE_ERROR. So at this
88  * point returning a value different from sz*nmemb indicates failure.
89  */
90  const size_t failure = bytes ? 0 : 1;
91 
92 #ifdef DEBUGBUILD
93  {
94  char *tty = curlx_getenv("CURL_ISATTY");
95  if(tty) {
96  is_tty = TRUE;
97  curl_free(tty);
98  }
99  }
100 
101  if(config->show_headers) {
102  if(bytes > (size_t)CURL_MAX_HTTP_HEADER) {
103  warnf(config->global, "Header data size exceeds single call write "
104  "limit!\n");
105  return failure;
106  }
107  }
108  else {
109  if(bytes > (size_t)CURL_MAX_WRITE_SIZE) {
110  warnf(config->global, "Data size exceeds single call write limit!\n");
111  return failure;
112  }
113  }
114 
115  {
116  /* Some internal congruency checks on received OutStruct */
117  bool check_fails = FALSE;
118  if(outs->filename) {
119  /* regular file */
120  if(!*outs->filename)
121  check_fails = TRUE;
122  if(!outs->s_isreg)
123  check_fails = TRUE;
124  if(outs->fopened && !outs->stream)
125  check_fails = TRUE;
126  if(!outs->fopened && outs->stream)
127  check_fails = TRUE;
128  if(!outs->fopened && outs->bytes)
129  check_fails = TRUE;
130  }
131  else {
132  /* standard stream */
133  if(!outs->stream || outs->s_isreg || outs->fopened)
134  check_fails = TRUE;
135  if(outs->alloc_filename || outs->is_cd_filename || outs->init)
136  check_fails = TRUE;
137  }
138  if(check_fails) {
139  warnf(config->global, "Invalid output struct data for write callback\n");
140  return failure;
141  }
142  }
143 #endif
144 
145  if(!outs->stream && !tool_create_output_file(outs, FALSE))
146  return failure;
147

168  wc_buf = (wchar_t*) malloc(wc_len * sizeof(wchar_t));
169  if(!wc_buf)
170  return failure;
171 
172  /* calculate buffer size for multi-byte characters */
173  wc_len = MultiByteToWideChar(CP_UTF8, 0, buffer, in_len, wc_buf, wc_len);
174  if(!wc_len) {
175  free(wc_buf);
176  return failure;
177  }
178 
179  fhnd = _get_osfhandle(fileno(outs->stream));
180 
181  if(!WriteConsoleW(
182  (HANDLE) fhnd,
183  wc_buf,
184  wc_len,
185  &wc_len,
186  NULL)) {
187  free(wc_buf);
188  return failure;
189  }
190  free(wc_buf);
191  rc = bytes;
192  }
193  else
194 #endif
195  rc = fwrite(buffer, sz, nmemb, outs->stream);
196 
197  if(bytes == rc)
198  /* we added this amount of data to the output */
199  outs->bytes += bytes;
200 
201  if(config->readbusy) {
202  config->readbusy = FALSE;
203  curl_easy_pause(config->easy, CURLPAUSE_CONT);
204  }
205 
206  if(config->nobuffer) {
207  /* output buffering disabled */
208  int res = fflush(outs->stream);
209  if(res)
210  return failure;
211  }
212 
213  return rc;
