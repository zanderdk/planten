49 EspFsInitResult espFsInit(void *flashAddress, size_t size) {
50 
51  spi_flash_init();
52  if (size % (64*1024) != 0) {
53  ESP_LOGE(tag, "Size is not divisible by 64K. Supplied was %d", size);
54  return ESPFS_INIT_RESULT_NO_IMAGE;
55  }
56  esp_err_t rc = spi_flash_mmap((uint32_t) flashAddress, size, SPI_FLASH_MMAP_DATA, (const void **)&espFlashPtr, &handle);
57  if (rc != ESP_OK) {
58  ESP_LOGD(tag, "rc from spi_flash_mmap: %d", rc);
59  }
60 
61  // check if there is valid header at address
62  EspFsHeader *testHeader = (EspFsHeader *)espFlashPtr;
63 
64  if (testHeader->magic != ESPFS_MAGIC) {
65  ESP_LOGE(tag, "No valid header at flash address. Expected to find %x and found %x", ESPFS_MAGIC, testHeader->magic);
66  return ESPFS_INIT_RESULT_NO_IMAGE;
67  }
68 
69  return ESPFS_INIT_RESULT_OK;
70 }

86 //Open a file and return a pointer to the file desc struct.
87 EspFsFile *espFsOpen(char *fileName) {
88  if (espFlashPtr == NULL) {
89  ESP_LOGD(tag, "Call espFsInit first!");
90  return NULL;
91  }
92  char *flashAddress = espFlashPtr;
93  char *hpos;
94  char *namebuf;
95  EspFsHeader *header;
96  EspFsFile *fileData;
97  //Strip initial slashes
98  while(fileName[0] == '/') {
99  fileName++;
100  }
101  //Go find that file!
102  while(1) {
103  hpos=flashAddress;
104  //Grab the next file header.
105  header = (EspFsHeader *)flashAddress;
106 
107  if (header->magic != ESPFS_MAGIC) {
108  ESP_LOGD(tag, "Magic mismatch. EspFS image broken.");
109  return NULL;
110  }
111  if (header->flags & FLAG_LASTFILE) {
112  ESP_LOGD(tag, "End of image. File not found.");
113  return NULL;
114  }
115  //Grab the name of the file.
116  flashAddress += sizeof(EspFsHeader);
117  namebuf = (char *)flashAddress;
118  if (strcmp(namebuf, fileName) == 0) {
119  //Yay, this is the file we need!
120  flashAddress += header->nameLen; //Skip to content.
121  fileData = (EspFsFile *)malloc(sizeof(EspFsFile)); //Alloc file desc mem
122  if (fileData==NULL) {
123  return NULL;
124  }
125  fileData->header = (EspFsHeader *)hpos;
126  fileData->decompressor = header->compression;
127  fileData->posComp = flashAddress;
128  fileData->posStart = flashAddress;
129  fileData->posDecomp = 0;
130  if (header->compression == COMPRESS_NONE) {
131  fileData->decompData = NULL;
132  } else {
133  ESP_LOGD(tag, "Invalid compression: %d", header->compression);
134  return NULL;
135  }
136  return fileData;
137  }
138  //We don't need this file. Skip name and file
139  flashAddress += header->nameLen+header->fileLenComp;
140  if ((int)flashAddress&3) {
141  flashAddress += 4-((int)flashAddress & 3); //align to next 32bit val
142  }
143  }
144 }

146 //Read len bytes from the given file into buff. Returns the actual amount of bytes read.
147 int espFsRead(EspFsFile *fh, char *buff, int len) {
148  int flen;
149  if (fh==NULL) {
150  return 0;
151  }
152 
153  memcpy((char*)&flen, (char*)&fh->header->fileLenComp, 4);
154 
155  if (fh->decompressor == COMPRESS_NONE) {
156  int toRead;
157  toRead = flen-(fh->posComp-fh->posStart);
158  if (len > toRead) {
159  len = toRead;
160  }
161  ESP_LOGD(tag, "copying %d bytes from 0x%x to 0x%x", len, (uint32_t)fh->posComp, (uint32_t)buff);
162  memcpy(buff, fh->posComp, len);
163  fh->posDecomp += len;
164  fh->posComp += len;
165  return len;
166  }
167  return 0;
168 }
