193 void HttpParser::parse(Socket s) {
194  ESP_LOGD(LOG_TAG, ">> parse: socket: %s", s.toString().c_str());
195  std::string line;
196  line = s.readToDelim(lineTerminator);
197  parseRequestLine(line);
198  line = s.readToDelim(lineTerminator);
199  while (!line.empty()) {
200  m_headers.insert(parseHeader(line));
201  line = s.readToDelim(lineTerminator);
202  }
203  // Only PUT and POST requests have a body
204  if (getMethod() != "POST" && getMethod() != "PUT") {
205  ESP_LOGD(LOG_TAG, "<< parse");
206  return;
207  }
208 
209  // We have now parsed up to and including the separator ... we are now at the point where we
210  // want to read the body. There are two stories here. The first is that we know the exact length
211  // of the body or we read until we can't read anymore.
212  if (hasHeader(HttpRequest::HTTP_HEADER_CONTENT_LENGTH)) {
213  std::string val = getHeader(HttpRequest::HTTP_HEADER_CONTENT_LENGTH);
214  int length = std::atoi(val.c_str());
215  uint8_t data[length];
216  s.receive(data, length, true);
217  m_body = std::string((char*) data, length);
218  } else {
219  uint8_t data[512];
220  int rc = s.receive(data, sizeof(data));
221  if (rc > 0) {
222  m_body = std::string((char*) data, rc);
223  }
224  }
225  ESP_LOGD(LOG_TAG, "<< parse: Size of body: %d", m_body.length());
226 } // parse
