154 /*
155  * ntlm_decode_type2_target()
156  *
157  * This is used to decode the "target info" in the NTLM type-2 message
158  * received.
159  *
160  * Parameters:
161  *
162  * data [in] - The session handle.
163  * buffer [in] - The decoded type-2 message.
164  * size [in] - The input buffer size, at least 32 bytes.
165  * ntlm [in/out] - The NTLM data struct being used and modified.
166  *
167  * Returns CURLE_OK on success.
168  */
169 static CURLcode ntlm_decode_type2_target(struct Curl_easy *data,
170  unsigned char *buffer,
171  size_t size,
172  struct ntlmdata *ntlm)
173 {
174  unsigned short target_info_len = 0;
175  unsigned int target_info_offset = 0;
176 
177 #if defined(CURL_DISABLE_VERBOSE_STRINGS)
178  (void) data;
179 #endif
180 
181  if(size >= 48) {
182  target_info_len = Curl_read16_le(&buffer[40]);
183  target_info_offset = Curl_read32_le(&buffer[44]);
184  if(target_info_len > 0) {
185  if(((target_info_offset + target_info_len) > size) ||
186  (target_info_offset < 48)) {
187  infof(data, "NTLM handshake failure (bad type-2 message). "
188  "Target Info Offset Len is set incorrect by the peer\n");
189  return CURLE_BAD_CONTENT_ENCODING;
190  }
191 
192  ntlm->target_info = malloc(target_info_len);
193  if(!ntlm->target_info)
194  return CURLE_OUT_OF_MEMORY;
195 
196  memcpy(ntlm->target_info, &buffer[target_info_offset], target_info_len);
197  }
198  }
199 
200  ntlm->target_info_len = target_info_len;
201 
202  return CURLE_OK;
203 }

205 /*
206  NTLM message structure notes:
207 
208  A 'short' is a 'network short', a little-endian 16-bit unsigned value.
209 
210  A 'long' is a 'network long', a little-endian, 32-bit unsigned value.
211 
212  A 'security buffer' represents a triplet used to point to a buffer,
213  consisting of two shorts and one long:
214 
215  1. A 'short' containing the length of the buffer content in bytes.
216  2. A 'short' containing the allocated space for the buffer in bytes.
217  3. A 'long' containing the offset to the start of the buffer in bytes,
218  from the beginning of the NTLM message.
219 */

235 /*
236  * Curl_auth_decode_ntlm_type2_message()
237  *
238  * This is used to decode an already encoded NTLM type-2 message. The message
239  * is first decoded from a base64 string into a raw NTLM message and checked
240  * for validity before the appropriate data for creating a type-3 message is
241  * written to the given NTLM data structure.
242  *
243  * Parameters:
244  *
245  * data [in] - The session handle.
246  * type2msg [in] - The base64 encoded type-2 message.
247  * ntlm [in/out] - The NTLM data struct being used and modified.
248  *
249  * Returns CURLE_OK on success.
250  */
251 CURLcode Curl_auth_decode_ntlm_type2_message(struct Curl_easy *data,
252  const char *type2msg,
253  struct ntlmdata *ntlm)
254 {
255  static const char type2_marker[] = { 0x02, 0x00, 0x00, 0x00 };
256 
257  /* NTLM type-2 message structure:
258 
259  Index Description Content
260  0 NTLMSSP Signature Null-terminated ASCII "NTLMSSP"
261  (0x4e544c4d53535000)
262  8 NTLM Message Type long (0x02000000)
263  12 Target Name security buffer
264  20 Flags long
265  24 Challenge 8 bytes
266  (32) Context 8 bytes (two consecutive longs) (*)
267  (40) Target Information security buffer (*)
268  (48) OS Version Structure 8 bytes (*)
269  32 (48) (56) Start of data block (*)
270  (*) -> Optional
271  */
272 
273  CURLcode result = CURLE_OK;
274  unsigned char *type2 = NULL;
275  size_t type2_len = 0;
276 
277 #if defined(NTLM_NEEDS_NSS_INIT)
278  /* Make sure the crypto backend is initialized */
279  result = Curl_nss_force_init(data);
280  if(result)
281  return result;
282 #elif defined(CURL_DISABLE_VERBOSE_STRINGS)
283  (void)data;
284 #endif
285 
286  /* Decode the base-64 encoded type-2 message */
287  if(strlen(type2msg) && *type2msg != '=') {
288  result = Curl_base64_decode(type2msg, &type2, &type2_len);
289  if(result)
290  return result;
291  }
292 
293  /* Ensure we have a valid type-2 message */
294  if(!type2) {
295  infof(data, "NTLM handshake failure (empty type-2 message)\n");
296  return CURLE_BAD_CONTENT_ENCODING;
297  }
298 
299  ntlm->flags = 0;
300 
301  if((type2_len < 32) ||
302  (memcmp(type2, NTLMSSP_SIGNATURE, 8) != 0) ||
303  (memcmp(type2 + 8, type2_marker, sizeof(type2_marker)) != 0)) {
304  /* This was not a good enough type-2 message */
305  free(type2);
306  infof(data, "NTLM handshake failure (bad type-2 message)\n");
307  return CURLE_BAD_CONTENT_ENCODING;
308  }
309 
310  ntlm->flags = Curl_read32_le(&type2[20]);
311  memcpy(ntlm->nonce, &type2[24], 8);
312 
313  if(ntlm->flags & NTLMFLAG_NEGOTIATE_TARGET_INFO) {
314  result = ntlm_decode_type2_target(data, type2, type2_len, ntlm);
315  if(result) {
316  free(type2);
317  infof(data, "NTLM handshake failure (bad type-2 message)\n");
318  return result;
319  }
320  }
321 
322  DEBUG_OUT({
323  fprintf(stderr, "**** TYPE2 header flags=0x%08.8lx ", ntlm->flags);
324  ntlm_print_flags(stderr, ntlm->flags);
325  fprintf(stderr, "\n nonce=");
326  ntlm_print_hex(stderr, (char *)ntlm->nonce, 8);
327  fprintf(stderr, "\n****\n");
328  fprintf(stderr, "**** Header %s\n ", header);
329  });
330 
331  free(type2);
332 
333  return result;
334 }

347 /*
348  * Curl_auth_create_ntlm_type1_message()
349  *
350  * This is used to generate an already encoded NTLM type-1 message ready for
351  * sending to the recipient using the appropriate compile time crypto API.
352  *
353  * Parameters:
354  *
355  * data [in] - The session handle.
356  * userp [in] - The user name in the format User or Domain\User.
357  * passwdp [in] - The user's password.
358  * service [in] - The service type such as http, smtp, pop or imap.
359  * host [in] - The host name.
360  * ntlm [in/out] - The NTLM data struct being used and modified.
361  * outptr [in/out] - The address where a pointer to newly allocated memory
362  * holding the result will be stored upon completion.
363  * outlen [out] - The length of the output message.
364  *
365  * Returns CURLE_OK on success.
366  */
367 CURLcode Curl_auth_create_ntlm_type1_message(struct Curl_easy *data,
368  const char *userp,
369  const char *passwdp,
370  const char *service,
371  const char *hostname,
372  struct ntlmdata *ntlm,
373  char **outptr, size_t *outlen)
374 {
375  /* NTLM type-1 message structure:
376 
377  Index Description Content
378  0 NTLMSSP Signature Null-terminated ASCII "NTLMSSP"
379  (0x4e544c4d53535000)
380  8 NTLM Message Type long (0x01000000)
381  12 Flags long
382  (16) Supplied Domain security buffer (*)
383  (24) Supplied Workstation security buffer (*)
384  (32) OS Version Structure 8 bytes (*)
385  (32) (40) Start of data block (*)
386  (*) -> Optional
387  */
388 
389  size_t size;
390 
391  unsigned char ntlmbuf[NTLM_BUFSIZE];
392  const char *host = ""; /* empty */
393  const char *domain = ""; /* empty */
394  size_t hostlen = 0;
395  size_t domlen = 0;
396  size_t hostoff = 0;
397  size_t domoff = hostoff + hostlen; /* This is 0: remember that host and
398  domain are empty */
399  (void)userp;
400  (void)passwdp;
401  (void)service,
402  (void)hostname,
403 
404  /* Clean up any former leftovers and initialise to defaults */
405  Curl_auth_ntlm_cleanup(ntlm);
406 
407 #if defined(USE_NTRESPONSES) && defined(USE_NTLM2SESSION)
408 #define NTLM2FLAG NTLMFLAG_NEGOTIATE_NTLM2_KEY
409 #else
410 #define NTLM2FLAG 0
411 #endif
412  msnprintf((char *)ntlmbuf, NTLM_BUFSIZE,
413  NTLMSSP_SIGNATURE "%c"
414  "\x01%c%c%c" /* 32-bit type = 1 */
415  "%c%c%c%c" /* 32-bit NTLM flag field */
416  "%c%c" /* domain length */
417  "%c%c" /* domain allocated space */
418  "%c%c" /* domain name offset */
419  "%c%c" /* 2 zeroes */
420  "%c%c" /* host length */
421  "%c%c" /* host allocated space */
422  "%c%c" /* host name offset */
423  "%c%c" /* 2 zeroes */
424  "%s" /* host name */
425  "%s", /* domain string */
426  0, /* trailing zero */
427  0, 0, 0, /* part of type-1 long */
428 
429  LONGQUARTET(NTLMFLAG_NEGOTIATE_OEM |
430  NTLMFLAG_REQUEST_TARGET |
431  NTLMFLAG_NEGOTIATE_NTLM_KEY |
432  NTLM2FLAG |
433  NTLMFLAG_NEGOTIATE_ALWAYS_SIGN),
434  SHORTPAIR(domlen),
435  SHORTPAIR(domlen),
436  SHORTPAIR(domoff),
437  0, 0,
438  SHORTPAIR(hostlen),
439  SHORTPAIR(hostlen),
440  SHORTPAIR(hostoff),
441  0, 0,
442  host, /* this is empty */
443  domain /* this is empty */);
444 
445  /* Initial packet length */
446  size = 32 + hostlen + domlen;
447 
448  DEBUG_OUT({
449  fprintf(stderr, "* TYPE1 header flags=0x%02.2x%02.2x%02.2x%02.2x "
450  "0x%08.8x ",
451  LONGQUARTET(NTLMFLAG_NEGOTIATE_OEM |
452  NTLMFLAG_REQUEST_TARGET |
453  NTLMFLAG_NEGOTIATE_NTLM_KEY |
454  NTLM2FLAG |
455  NTLMFLAG_NEGOTIATE_ALWAYS_SIGN),
456  NTLMFLAG_NEGOTIATE_OEM |
457  NTLMFLAG_REQUEST_TARGET |
458  NTLMFLAG_NEGOTIATE_NTLM_KEY |
459  NTLM2FLAG |
460  NTLMFLAG_NEGOTIATE_ALWAYS_SIGN);
461  ntlm_print_flags(stderr,
462  NTLMFLAG_NEGOTIATE_OEM |
463  NTLMFLAG_REQUEST_TARGET |
464  NTLMFLAG_NEGOTIATE_NTLM_KEY |
465  NTLM2FLAG |
466  NTLMFLAG_NEGOTIATE_ALWAYS_SIGN);
467  fprintf(stderr, "\n****\n");
468  });
469 
470  /* Return with binary blob encoded into base64 */
471  return Curl_base64_encode(data, (char *)ntlmbuf, size, outptr, outlen);
472 }

474 /*
475  * Curl_auth_create_ntlm_type3_message()
476  *
477  * This is used to generate an already encoded NTLM type-3 message ready for
478  * sending to the recipient using the appropriate compile time crypto API.
479  *
480  * Parameters:
481  *
482  * data [in] - The session handle.
483  * userp [in] - The user name in the format User or Domain\User.
484  * passwdp [in] - The user's password.
485  * ntlm [in/out] - The NTLM data struct being used and modified.
486  * outptr [in/out] - The address where a pointer to newly allocated memory
487  * holding the result will be stored upon completion.
488  * outlen [out] - The length of the output message.
489  *
490  * Returns CURLE_OK on success.
491  */
492 CURLcode Curl_auth_create_ntlm_type3_message(struct Curl_easy *data,
493  const char *userp,
494  const char *passwdp,
495  struct ntlmdata *ntlm,
496  char **outptr, size_t *outlen)
497 
498 {
499  /* NTLM type-3 message structure:
500 
501  Index Description Content
502  0 NTLMSSP Signature Null-terminated ASCII "NTLMSSP"
503  (0x4e544c4d53535000)
504  8 NTLM Message Type long (0x03000000)
505  12 LM/LMv2 Response security buffer
506  20 NTLM/NTLMv2 Response security buffer
507  28 Target Name security buffer
508  36 User Name security buffer
509  44 Workstation Name security buffer
510  (52) Session Key security buffer (*)
511  (60) Flags long (*)
512  (64) OS Version Structure 8 bytes (*)
513  52 (64) (72) Start of data block
514  (*) -> Optional
515  */
516 
517  CURLcode result = CURLE_OK;
518  size_t size;
519  unsigned char ntlmbuf[NTLM_BUFSIZE];
520  int lmrespoff;
521  unsigned char lmresp[24]; /* fixed-size */
522 #ifdef USE_NTRESPONSES
523  int ntrespoff;
524  unsigned int ntresplen = 24;
525  unsigned char ntresp[24]; /* fixed-size */
526  unsigned char *ptr_ntresp = &ntresp[0];
527  unsigned char *ntlmv2resp = NULL;
528 #endif
529  bool unicode = (ntlm->flags & NTLMFLAG_NEGOTIATE_UNICODE) ? TRUE : FALSE;
530  char host[HOSTNAME_MAX + 1] = "";
531  const char *user;
532  const char *domain = "";
533  size_t hostoff = 0;
534  size_t useroff = 0;
535  size_t domoff = 0;
536  size_t hostlen = 0;
537  size_t userlen = 0;
538  size_t domlen = 0;
539 
540  user = strchr(userp, '\\');
541  if(!user)
542  user = strchr(userp, '/');
543 
544  if(user) {
545  domain = userp;
546  domlen = (user - domain);
547  user++;
548  }
549  else
550  user = userp;
551 
552  userlen = strlen(user);
553 
554  /* Get the machine's un-qualified host name as NTLM doesn't like the fully
555  qualified domain name */
556  if(Curl_gethostname(host, sizeof(host))) {
557  infof(data, "gethostname() failed, continuing without!\n");
558  hostlen = 0;
559  }
560  else {
561  hostlen = strlen(host);
562  }
563 
564 #if defined(USE_NTRESPONSES) && defined(USE_NTLM_V2)
565  if(ntlm->target_info_len) {
566  unsigned char ntbuffer[0x18];
567  unsigned char entropy[8];
568  unsigned char ntlmv2hash[0x18];
569 
570  result = Curl_rand(data, entropy, 8);
571  if(result)
572  return result;
573 
574  result = Curl_ntlm_core_mk_nt_hash(data, passwdp, ntbuffer);
575  if(result)
576  return result;
577 
578  result = Curl_ntlm_core_mk_ntlmv2_hash(user, userlen, domain, domlen,
579  ntbuffer, ntlmv2hash);
580  if(result)
581  return result;
582 
583  /* LMv2 response */
584  result = Curl_ntlm_core_mk_lmv2_resp(ntlmv2hash, entropy,
585  &ntlm->nonce[0], lmresp);
586  if(result)
587  return result;
588 
589  /* NTLMv2 response */
590  result = Curl_ntlm_core_mk_ntlmv2_resp(ntlmv2hash, entropy,
591  ntlm, &ntlmv2resp, &ntresplen);
592  if(result)
593  return result;
594 
595  ptr_ntresp = ntlmv2resp;
596  }
597  else
598 #endif
599 
600 #if defined(USE_NTRESPONSES) && defined(USE_NTLM2SESSION)
601  /* We don't support NTLM2 if we don't have USE_NTRESPONSES */
602  if(ntlm->flags & NTLMFLAG_NEGOTIATE_NTLM2_KEY) {
603  unsigned char ntbuffer[0x18];
604  unsigned char tmp[0x18];
605  unsigned char md5sum[MD5_DIGEST_LENGTH];
