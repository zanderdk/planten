79 MAX7219::MAX7219(SPI* spi, int numDevices) {
80  assert(spi != nullptr);
81  this->spi = spi;
82  if (numDevices <= 0 || numDevices > 8) {
83  numDevices = 8;
84  }
85  maxDevices = numDevices;
86 
87  for (uint8_t i = 0; i < 64; i++) {
88  status[i] = 0x00;
89  }
90  for (int i = 0; i < maxDevices; i++) {
91  spiTransfer(i, OP_DISPLAYTEST, 0);
92  //scanlimit is set to max on startup
93  setScanLimit(7, i);
94  //decode is done in source
95  spiTransfer(i, OP_DECODEMODE, 0);
96  clearDisplay(i);
97  //we go into shutdown-mode on startup
98  shutdown(true, i);
99  }
100 }

147 void MAX7219::setLed(int row, int column, bool state, int addr) {
148  if (addr < 0 || addr >= maxDevices) return;
149 
150  int offset;
151  uint8_t val = 0;
152 
153  if (row < 0 || row > 7 || column < 0 || column > 7) {
154  return;
155  }
156  offset = addr * 8;
157  val = 0b10000000 >> column;
158  if (state) {
159  status[offset + row] = status[offset + row] | val;
160  } else {
161  val = ~val;
162  status[offset + row] = status[offset + row] & val;
163  }
164  spiTransfer(addr, row + 1, status[offset + row]);
165 }

191 void MAX7219::setDigit(int digit, uint8_t value, bool dp, int addr) {
192  if (addr < 0 || addr >= maxDevices) return;
193  if (digit < 0 || digit > 7 || value > 15) return;
194 
195  int offset = addr * 8;

205 void MAX7219::setChar(int digit, char value, bool dp, int addr) {
206  if (addr < 0 || addr >= maxDevices) return;
207  if (digit < 0 || digit > 7) return;
208 
209  int offset = addr * 8;
210  uint8_t index = (uint8_t) value;
211  if (index > 127) index = 32; // not defined beyond index 127, so we use the space char
212  uint8_t v = charTable[index];
213  if (dp) {
214  v |= 0b10000000;
215  }
216  status[offset + digit] = v;
217  spiTransfer(addr, digit + 1, v);
218 }

221 void MAX7219::spiTransfer(int addr, volatile uint8_t opcode, volatile uint8_t data) {
222  //Create an array with the data to shift out
223  int offset = addr * 2;
224  int maxbytes = maxDevices * 2;
225 
226  /* The array for shifting the data to the devices */
227  uint8_t spidata[16];
228 
229  for (int i = 0; i < maxbytes; i++) {
230  spidata[i] = (uint8_t) 0;
231  }
232  //put our device data into the array
233  spidata[offset] = opcode;
234  spidata[offset + 1] = data;
235  spi->transfer(spidata, maxbytes);
236 }
