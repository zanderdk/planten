45 /* Returns timeout in ms. 0 or negative number means the timeout has already
46  triggered */
47 time_t Curl_pp_state_timeout(struct pingpong *pp)
48 {
49  struct connectdata *conn = pp->conn;
50  struct Curl_easy *data = conn->data;
51  time_t timeout_ms; /* in milliseconds */
52  long response_time = (data->set.server_response_timeout)?
53  data->set.server_response_timeout: pp->response_time;
54 
55  /* if CURLOPT_SERVER_RESPONSE_TIMEOUT is set, use that to determine
56  remaining time, or use pp->response because SERVER_RESPONSE_TIMEOUT is
57  supposed to govern the response for any given server response, not for
58  the time from connect to the given server response. */
59 
60  /* Without a requested timeout, we only wait 'response_time' seconds for the
61  full response to arrive before we bail out */
62  timeout_ms = response_time -
63  Curl_timediff(Curl_now(), pp->response); /* spent time */
64 
65  if(data->set.timeout) {
66  /* if timeout is requested, find out how much remaining time we have */
67  time_t timeout2_ms = data->set.timeout - /* timeout time */
68  Curl_timediff(Curl_now(), conn->now); /* spent time */
69 
70  /* pick the lowest number */
71  timeout_ms = CURLMIN(timeout_ms, timeout2_ms);
72  }
73 
74  return timeout_ms;
75 }

77 /*
78  * Curl_pp_statemach()
79  */
80 CURLcode Curl_pp_statemach(struct pingpong *pp, bool block)
81 {
82  struct connectdata *conn = pp->conn;
83  curl_socket_t sock = conn->sock[FIRSTSOCKET];
84  int rc;
85  time_t interval_ms;
86  time_t timeout_ms = Curl_pp_state_timeout(pp);
87  struct Curl_easy *data = conn->data;
88  CURLcode result = CURLE_OK;
89 
90  if(timeout_ms <= 0) {
91  failf(data, "server response timeout");
92  return CURLE_OPERATION_TIMEDOUT; /* already too little time */
93  }
94 
95  if(block) {
96  interval_ms = 1000; /* use 1 second timeout intervals */
97  if(timeout_ms < interval_ms)
98  interval_ms = timeout_ms;
99  }
100  else
101  interval_ms = 0; /* immediate */
102 
103  if(Curl_ssl_data_pending(conn, FIRSTSOCKET))
104  rc = 1;
105  else if(Curl_pp_moredata(pp))
106  /* We are receiving and there is data in the cache so just read it */
107  rc = 1;
108  else if(!pp->sendleft && Curl_ssl_data_pending(conn, FIRSTSOCKET))
109  /* We are receiving and there is data ready in the SSL library */
110  rc = 1;
111  else
112  rc = Curl_socket_check(pp->sendleft?CURL_SOCKET_BAD:sock, /* reading */
113  CURL_SOCKET_BAD,
114  pp->sendleft?sock:CURL_SOCKET_BAD, /* writing */
115  interval_ms);
116 
117  if(block) {
118  /* if we didn't wait, we don't have to spend time on this now */
119  if(Curl_pgrsUpdate(conn))
120  result = CURLE_ABORTED_BY_CALLBACK;
121  else
122  result = Curl_speedcheck(data, Curl_now());
123 
124  if(result)
125  return result;
126  }
127 
128  if(rc == -1) {
129  failf(data, "select/poll error");
130  result = CURLE_OUT_OF_MEMORY;
131  }
132  else if(rc)
133  result = pp->statemach_act(conn);
134 
135  return result;
136 }

150 /***********************************************************************
151  *
152  * Curl_pp_vsendf()
153  *
154  * Send the formatted string as a command to a pingpong server. Note that
155  * the string should not have any CRLF appended, as this function will
156  * append the necessary things itself.
157  *
158  * made to never block
159  */
160 CURLcode Curl_pp_vsendf(struct pingpong *pp,
161  const char *fmt,
162  va_list args)
163 {
164  ssize_t bytes_written;
165  size_t write_len;
166  char *fmt_crlf;
167  char *s;
168  CURLcode result;
169  struct connectdata *conn = pp->conn;
170  struct Curl_easy *data;
171 
172 #ifdef HAVE_GSSAPI
173  enum protection_level data_sec;
174 #endif
175 
176  DEBUGASSERT(pp->sendleft == 0);
177  DEBUGASSERT(pp->sendsize == 0);
178  DEBUGASSERT(pp->sendthis == NULL);
179 
180  if(!conn)
181  /* can't send without a connection! */
182  return CURLE_SEND_ERROR;
183 
184  data = conn->data;
185 
186  fmt_crlf = aprintf("%s\r\n", fmt); /* append a trailing CRLF */
187  if(!fmt_crlf)
188  return CURLE_OUT_OF_MEMORY;
189 
190  s = vaprintf(fmt_crlf, args); /* trailing CRLF appended */
191  free(fmt_crlf);
192  if(!s)
193  return CURLE_OUT_OF_MEMORY;
194 
195  bytes_written = 0;
196  write_len = strlen(s);
197 
198  Curl_pp_init(pp);
199 
200  result = Curl_convert_to_network(data, s, write_len);
201  /* Curl_convert_to_network calls failf if unsuccessful */
202  if(result) {
203  free(s);
204  return result;
205  }
206 
207 #ifdef HAVE_GSSAPI
208  conn->data_prot = PROT_CMD;
209 #endif
210  result = Curl_write(conn, conn->sock[FIRSTSOCKET], s, write_len,
211  &bytes_written);
212 #ifdef HAVE_GSSAPI
213  data_sec = conn->data_prot;
214  DEBUGASSERT(data_sec > PROT_NONE && data_sec < PROT_LAST);
215  conn->data_prot = data_sec;
216 #endif
217 
218  if(result) {
219  free(s);
220  return result;
221  }
222 
223  if(conn->data->set.verbose)
224  Curl_debug(conn->data, CURLINFO_HEADER_OUT, s, (size_t)bytes_written);
225 
226  if(bytes_written != (ssize_t)write_len) {
227  /* the whole chunk was not sent, keep it around and adjust sizes */
228  pp->sendthis = s;
229  pp->sendsize = write_len;
230  pp->sendleft = write_len - bytes_written;
231  }
232  else {
233  free(s);
234  pp->sendthis = NULL;
235  pp->sendleft = pp->sendsize = 0;
236  pp->response = Curl_now();
237  }
238 
239  return CURLE_OK;
240 }

268  * Curl_pp_readresp()
269  *
270  * Reads a piece of a server response.
271  */
272 CURLcode Curl_pp_readresp(curl_socket_t sockfd,
273  struct pingpong *pp,
274  int *code, /* return the server code if done */
275  size_t *size) /* size of the response */
276 {
277  ssize_t perline; /* count bytes per line */
278  bool keepon = TRUE;
279  ssize_t gotbytes;
280  char *ptr;
281  struct connectdata *conn = pp->conn;
282  struct Curl_easy *data = conn->data;
283  char * const buf = data->state.buffer;
284  CURLcode result = CURLE_OK;
285 
286  *code = 0; /* 0 for errors or not done */
287  *size = 0;
288 
289  ptr = buf + pp->nread_resp;
290 
291  /* number of bytes in the current line, so far */
292  perline = (ssize_t)(ptr-pp->linestart_resp);
293 
294  while((pp->nread_resp < (size_t)data->set.buffer_size) &&
295  (keepon && !result)) {
296 
297  if(pp->cache) {
298  /* we had data in the "cache", copy that instead of doing an actual
299  * read
300  *
301  * pp->cache_size is cast to ssize_t here. This should be safe, because
302  * it would have been populated with something of size int to begin
303  * with, even though its datatype may be larger than an int.
304  */
305  if((ptr + pp->cache_size) > (buf + data->set.buffer_size + 1)) {
306  failf(data, "cached response data too big to handle");
307  return CURLE_RECV_ERROR;
308  }
309  memcpy(ptr, pp->cache, pp->cache_size);
310  gotbytes = (ssize_t)pp->cache_size;
311  free(pp->cache); /* free the cache */
312  pp->cache = NULL; /* clear the pointer */
313  pp->cache_size = 0; /* zero the size just in case */
314  }
315  else {
316 #ifdef HAVE_GSSAPI
317  enum protection_level prot = conn->data_prot;
318  conn->data_prot = PROT_CLEAR;
319 #endif
320  DEBUGASSERT((ptr + data->set.buffer_size - pp->nread_resp) <=
321  (buf + data->set.buffer_size + 1));
322  result = Curl_read(conn, sockfd, ptr,
323  data->set.buffer_size - pp->nread_resp,
324  &gotbytes);
325 #ifdef HAVE_GSSAPI
326  DEBUGASSERT(prot > PROT_NONE && prot < PROT_LAST);
327  conn->data_prot = prot;
328 #endif
329  if(result == CURLE_AGAIN)
330  return CURLE_OK; /* return */
331 
332  if(!result && (gotbytes > 0))
333  /* convert from the network encoding */
334  result = Curl_convert_from_network(data, ptr, gotbytes);
335  /* Curl_convert_from_network calls failf if unsuccessful */
336 
337  if(result)
338  /* Set outer result variable to this error. */
339  keepon = FALSE;
340  }
341 
342  if(!keepon)
343  ;
344  else if(gotbytes <= 0) {
345  keepon = FALSE;
346  result = CURLE_RECV_ERROR;
347  failf(data, "response reading failed");
348  }
349  else {
350  /* we got a whole chunk of data, which can be anything from one
351  * byte to a set of lines and possible just a piece of the last
352  * line */
353  ssize_t i;
354  ssize_t clipamount = 0;
355  bool restart = FALSE;
356 
357  data->req.headerbytecount += (long)gotbytes;
358 
359  pp->nread_resp += gotbytes;
360  for(i = 0; i < gotbytes; ptr++, i++) {
361  perline++;
362  if(*ptr == '\n') {
363  /* a newline is CRLF in pp-talk, so the CR is ignored as
364  the line isn't really terminated until the LF comes */
365 
366  /* output debug output if that is requested */
367 #ifdef HAVE_GSSAPI
368  if(!conn->sec_complete)
369 #endif
370  if(data->set.verbose)
371  Curl_debug(data, CURLINFO_HEADER_IN,
372  pp->linestart_resp, (size_t)perline);
373 
374  /*
375  * We pass all response-lines to the callback function registered
376  * for "headers". The response lines can be seen as a kind of
377  * headers.
378  */
379  result = Curl_client_write(conn, CLIENTWRITE_HEADER,
380  pp->linestart_resp, perline);
381  if(result)
382  return result;
383 
384  if(pp->endofresp(conn, pp->linestart_resp, perline, code)) {
385  /* This is the end of the last line, copy the last line to the
386  start of the buffer and zero terminate, for old times sake */
387  size_t n = ptr - pp->linestart_resp;
388  memmove(buf, pp->linestart_resp, n);
389  buf[n] = 0; /* zero terminate */
390  keepon = FALSE;
391  pp->linestart_resp = ptr + 1; /* advance pointer */
392  i++; /* skip this before getting out */
393 
394  *size = pp->nread_resp; /* size of the response */
395  pp->nread_resp = 0; /* restart */
396  break;
397  }
398  perline = 0; /* line starts over here */
399  pp->linestart_resp = ptr + 1;
400  }
401  }
402 
403  if(!keepon && (i != gotbytes)) {
404  /* We found the end of the response lines, but we didn't parse the
405  full chunk of data we have read from the server. We therefore need
406  to store the rest of the data to be checked on the next invoke as
407  it may actually contain another end of response already! */
408  clipamount = gotbytes - i;
409  restart = TRUE;
410  DEBUGF(infof(data, "Curl_pp_readresp_ %d bytes of trailing "
411  "server response left\n",
412  (int)clipamount));
413  }
414  else if(keepon) {
415 
416  if((perline == gotbytes) && (gotbytes > data->set.buffer_size/2)) {
417  /* We got an excessive line without newlines and we need to deal
418  with it. We keep the first bytes of the line then we throw
419  away the rest. */
420  infof(data, "Excessive server response line length received, "
421  "%zd bytes. Stripping\n", gotbytes);
422  restart = TRUE;
423 
424  /* we keep 40 bytes since all our pingpong protocols are only
425  interested in the first piece */
426  clipamount = 40;
427  }
428  else if(pp->nread_resp > (size_t)data->set.buffer_size/2) {
429  /* We got a large chunk of data and there's potentially still
430  trailing data to take care of, so we put any such part in the
431  "cache", clear the buffer to make space and restart. */
432  clipamount = perline;
433  restart = TRUE;
434  }
435  }
436  else if(i == gotbytes)
437  restart = TRUE;
438 
439  if(clipamount) {
440  pp->cache_size = clipamount;
441  pp->cache = malloc(pp->cache_size);
442  if(pp->cache)
443  memcpy(pp->cache, pp->linestart_resp, pp->cache_size);
444  else
445  return CURLE_OUT_OF_MEMORY;
446  }
447  if(restart) {
448  /* now reset a few variables to start over nicely from the start of
449  the big buffer */
450  pp->nread_resp = 0; /* start over from scratch in the buffer */
451  ptr = pp->linestart_resp = buf;
452  perline = 0;
453  }
454 
455  } /* there was data */
456 
457  } /* while there's buffer left and loop is requested */
458 
459  pp->pending_resp = FALSE;
460 
461  return result;
462 }

485 {
