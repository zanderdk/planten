106 static int
107 convert(char * d, size_t dlen, int dccsid,
108  const char * s, int slen, int sccsid)
109 
110 {
111  int i;
112  iconv_t cd;
113  size_t lslen;
114 
123  if(sccsid == 65535)
124  sccsid = ASCII_CCSID;
125 
126  if(dccsid == 65535)
127  dccsid = ASCII_CCSID;
128 
129  if(sccsid == dccsid) {
130  lslen = slen >= 0? slen: strlen(s) + 1;
131  i = lslen < dlen? lslen: dlen;
132 
133  if(s != d && i > 0)
134  memcpy(d, s, i);
135 
136  return i;
137  }
138 
139  if(slen < 0) {
140  lslen = 0;
141  cd = iconv_open_CCSID(dccsid, sccsid, 1);
142  }
143  else {
144  lslen = (size_t) slen;
145  cd = iconv_open_CCSID(dccsid, sccsid, 0);
146  }
147 
148  if(ICONV_OPEN_ERROR(cd))
149  return -1;
150 
151  i = dlen;
152 
153  if((int) iconv(cd, (char * *) &s, &lslen, &d, &dlen) < 0)
154  i = -1;
155  else
156  i -= dlen;
157 
158  iconv_close(cd);
159  return i;
160 }

163 static char *
164 dynconvert(int dccsid, const char * s, int slen, int sccsid)
165 
166 {
167  char * d;
168  char * cp;
169  size_t dlen;
170  int l;
171  static const char nullbyte = 0;
172 
173  /* Like convert, but the destination is allocated and returned. */
174 
175  dlen = (size_t) (slen < 0? strlen(s): slen) + 1;
176  dlen *= MAX_CONV_EXPANSION; /* Allow some expansion. */
177  d = malloc(dlen);
178 
179  if(!d)
180  return (char *) NULL;
181 
182  l = convert(d, dlen, dccsid, s, slen, sccsid);
183 
184  if(l < 0) {
185  free(d);
186  return (char *) NULL;
187  }
188 
189  if(slen < 0) {
190  /* Need to null-terminate even when source length is given.
191  Since destination code size is unknown, use a conversion to generate
192  terminator. */
193 
194  int l2 = convert(d + l, dlen - l, dccsid, &nullbyte, -1, ASCII_CCSID);
195 
196  if(l2 < 0) {
197  free(d);
198  return (char *) NULL;
199  }
200 
201  l += l2;
202  }
203 
204  if((size_t) l < dlen) {
205  cp = realloc(d, l); /* Shorten to minimum needed. */
206 
207  if(cp)
208  d = cp;
209  }
210 
211  return d;
212 }

370 convert_version_info_string(const char * * stringp,
371  char * * bufp, int * left, unsigned int ccsid)
372 
373 {
374  /* Helper for curl_version_info_ccsid(): convert a string if defined.
375  Result is stored in the `*left'-byte buffer at `*bufp'.
376  `*bufp' and `*left' are updated accordingly.
377  Return 0 if ok, else -1. */
378 
379  if(*stringp) {

381 
382  if(l <= 0)

394 curl_version_info_data *
395 curl_version_info_ccsid(CURLversion stamp, unsigned int ccsid)
396 
397 {
398  curl_version_info_data * p;
399  char * cp;
400  int n;
401  int nproto;
402  curl_version_info_data * id;
403 
404  /* The assertion below is possible, because although the second operand
405  is an enum member, the first is a #define. In that case, the OS/400 C
406  compiler seems to compare string values after substitution. */
407 
408 #if CURLVERSION_NOW != CURLVERSION_FOURTH
409 #error curl_version_info_data structure has changed: upgrade this procedure.
410 #endif
411 
412  /* If caller has been compiled with a new version, error. */
413 
414  if(stamp > CURLVERSION_NOW)
415  return (curl_version_info_data *) NULL;
416 
417  p = curl_version_info(stamp);
418 
419  if(!p)
420  return p;
421 
422  /* Measure thread space needed. */
423 
424  n = 0;
425  nproto = 0;
426 
427  if(p->protocols) {
428  while(p->protocols[nproto])
429  n += strlen(p->protocols[nproto++]);
430 
431  n += nproto++;
432  }
433 
434  if(p->version)
435  n += strlen(p->version) + 1;
436 
437  if(p->host)
438  n += strlen(p->host) + 1;
439 
440  if(p->ssl_version)
441  n += strlen(p->ssl_version) + 1;
442 
443  if(p->libz_version)
444  n += strlen(p->libz_version) + 1;
445 
446  if(p->ares)
447  n += strlen(p->ares) + 1;
448 
449  if(p->libidn)
450  n += strlen(p->libidn) + 1;
451 
452  if(p->libssh_version)
453  n += strlen(p->libssh_version) + 1;
454 
455  /* Allocate thread space. */
456 
457  n *= MAX_CONV_EXPANSION;
458 
459  if(nproto)
460  n += nproto * sizeof(const char *);
461 
462  cp = Curl_thread_buffer(LK_VERSION_INFO_DATA, n);
463  id = (curl_version_info_data *) Curl_thread_buffer(LK_VERSION_INFO,
464  sizeof *id);
465 
466  if(!id || !cp)
467  return (curl_version_info_data *) NULL;
468 
469  /* Copy data and convert strings. */
470 
471  memcpy((char *) id, (char *) p, sizeof *p);
472 
473  if(id->protocols) {
474  int i = nproto * sizeof id->protocols[0];
475 
476  id->protocols = (const char * const *) cp;
477  memcpy(cp, (char *) p->protocols, i);
478  cp += i;
479  n -= i;
480 
481  for(i = 0; id->protocols[i]; i++)
482  if(convert_version_info_string(((const char * *) id->protocols) + i,
483  &cp, &n, ccsid))
484  return (curl_version_info_data *) NULL;
485  }
486 
487  if(convert_version_info_string(&id->version, &cp, &n, ccsid))
488  return (curl_version_info_data *) NULL;
489 
490  if(convert_version_info_string(&id->host, &cp, &n, ccsid))
491  return (curl_version_info_data *) NULL;
492 
493  if(convert_version_info_string(&id->ssl_version, &cp, &n, ccsid))
494  return (curl_version_info_data *) NULL;
495 
496  if(convert_version_info_string(&id->libz_version, &cp, &n, ccsid))
497  return (curl_version_info_data *) NULL;
498 
499  if(convert_version_info_string(&id->ares, &cp, &n, ccsid))
500  return (curl_version_info_data *) NULL;
501 
502  if(convert_version_info_string(&id->libidn, &cp, &n, ccsid))
503  return (curl_version_info_data *) NULL;
504 
505  if(convert_version_info_string(&id->libssh_version, &cp, &n, ccsid))
506  return (curl_version_info_data *) NULL;
507 
508  return id;
509 }

605 CURLcode
606 curl_easy_getinfo_ccsid(CURL * curl, CURLINFO info, ...)
607 
608 {
609  va_list arg;
610  void * paramp;
611  CURLcode ret;
612  unsigned int ccsid;
613  char * * cpp;
614  struct Curl_easy * data;
615  struct curl_slist * * slp;
616  struct curl_certinfo * cipf;
617  struct curl_certinfo * cipt;
618 
619  /* WARNING: unlike curl_easy_getinfo(), the strings returned by this
620  procedure have to be free'ed. */
621 
622  data = (struct Curl_easy *) curl;
623  va_start(arg, info);
624  paramp = va_arg(arg, void *);
625  ret = Curl_getinfo(data, info, paramp);
626 
627  if(ret == CURLE_OK)
628  switch ((int) info & CURLINFO_TYPEMASK) {
629 
630  case CURLINFO_STRING:
631  ccsid = va_arg(arg, unsigned int);
632  cpp = (char * *) paramp;
633 
634  if(*cpp) {
635  *cpp = dynconvert(ccsid, *cpp, -1, ASCII_CCSID);
636 
637  if(!*cpp)
638  ret = CURLE_OUT_OF_MEMORY;
639  }
640 
641  break;
642 
643  case CURLINFO_SLIST:
644  ccsid = va_arg(arg, unsigned int);
645  switch (info) {
646  case CURLINFO_CERTINFO:
647  cipf = *(struct curl_certinfo * *) paramp;
648  if(cipf) {
649  if(!(cipt = (struct curl_certinfo *) malloc(sizeof *cipt)))
650  ret = CURLE_OUT_OF_MEMORY;
651  else {
652  cipt->certinfo = (struct curl_slist * *)
653  calloc(cipf->num_of_certs +
654  1, sizeof(struct curl_slist *));
655  if(!cipt->certinfo)
656  ret = CURLE_OUT_OF_MEMORY;
657  else {
658  int i;
659 
660  cipt->num_of_certs = cipf->num_of_certs;
661  for(i = 0; i < cipf->num_of_certs; i++)
662  if(cipf->certinfo[i])
663  if(!(cipt->certinfo[i] = slist_convert(ccsid,
664  cipf->certinfo[i],
665  ASCII_CCSID))) {
666  ret = CURLE_OUT_OF_MEMORY;
667  break;
668  }
669  }
670  }
671 
672  if(ret != CURLE_OK) {
673  curl_certinfo_free_all(cipt);
674  cipt = (struct curl_certinfo *) NULL;
675  }
676 
677  *(struct curl_certinfo * *) paramp = cipt;
678  }
679 
680  break;
681 
682  case CURLINFO_TLS_SESSION:
683  case CURLINFO_TLS_SSL_PTR:
684  case CURLINFO_SOCKET:
685  break;
686 
687  default:
688  slp = (struct curl_slist * *) paramp;
689  if(*slp)
690  if(!(*slp = slist_convert(ccsid, *slp, ASCII_CCSID)))
691  ret = CURLE_OUT_OF_MEMORY;
692  break;
693  }
694  }
695 
696  va_end(arg);
697  return ret;
698 }

779 CURLFORMcode
780 curl_formadd_ccsid(struct curl_httppost * * httppost,
781  struct curl_httppost * * last_post, ...)
782 
783 {
784  va_list arg;
785  CURLformoption option;
786  CURLFORMcode result;
787  struct curl_forms * forms;
788  struct curl_forms * lforms;
789  struct curl_forms * tforms;
790  unsigned int lformlen;
791  const char * value;
792  unsigned int ccsid;
793  int nargs;
794  int namex;
795  int namelengthx;
796  int contentx;
797  int lengthx;
798  unsigned int contentccsid;
799  unsigned int nameccsid;
800 
801  /* A single curl_formadd() call cannot be split in several calls to deal
802  with all parameters: the original parameters are thus copied to a local
803  curl_forms array and converted to ASCII when needed.
804  CURLFORM_PTRNAME is processed as if it were CURLFORM_COPYNAME.
805  CURLFORM_COPYNAME and CURLFORM_NAMELENGTH occurrence order in
806  parameters is not defined; for this reason, the actual conversion is
807  delayed to the end of parameter processing. The same applies to
808  CURLFORM_COPYCONTENTS/CURLFORM_CONTENTSLENGTH, but these may appear
809  several times in the parameter list; the problem resides here in knowing
810  which CURLFORM_CONTENTSLENGTH applies to which CURLFORM_COPYCONTENTS and
811  when we can be sure to have both info for conversion: end of parameter
812  list is such a point, but CURLFORM_CONTENTTYPE is also used here as a
813  natural separator between content data definitions; this seems to be
814  in accordance with FormAdd() behavior. */
815 
816  /* Allocate the local curl_forms array. */
817 
818  lformlen = ALLOC_GRANULE;
