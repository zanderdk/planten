104 /*
105  * Curl_sasl_decode_mech()
106  *
107  * Convert a SASL mechanism name into a token.
108  *
109  * Parameters:
110  *
111  * ptr [in] - The mechanism string.
112  * maxlen [in] - Maximum mechanism string length.
113  * len [out] - If not NULL, effective name length.
114  *
115  * Returns the SASL mechanism token or 0 if no match.
116  */
117 unsigned int Curl_sasl_decode_mech(const char *ptr, size_t maxlen, size_t *len)
118 {
119  unsigned int i;
120  char c;
121 
122  for(i = 0; mechtable[i].name; i++) {
123  if(maxlen >= mechtable[i].len &&
124  !memcmp(ptr, mechtable[i].name, mechtable[i].len)) {
125  if(len)
126  *len = mechtable[i].len;
127 
128  if(maxlen == mechtable[i].len)
129  return mechtable[i].bit;
130 
131  c = ptr[mechtable[i].len];
132  if(!ISUPPER(c) && !ISDIGIT(c) && c != '-' && c != '_')
133  return mechtable[i].bit;
134  }
135  }
136 
137  return 0;
138 }

140 /*
141  * Curl_sasl_parse_url_auth_option()
142  *
143  * Parse the URL login options.
144  */
145 CURLcode Curl_sasl_parse_url_auth_option(struct SASL *sasl,
146  const char *value, size_t len)
147 {
148  CURLcode result = CURLE_OK;
149  size_t mechlen;
150 
151  if(!len)
152  return CURLE_URL_MALFORMAT;
153 
154  if(sasl->resetprefs) {
155  sasl->resetprefs = FALSE;
156  sasl->prefmech = SASL_AUTH_NONE;
157  }
158 
159  if(!strncmp(value, "*", len))
160  sasl->prefmech = SASL_AUTH_DEFAULT;
161  else {
162  unsigned int mechbit = Curl_sasl_decode_mech(value, len, &mechlen);
163  if(mechbit && mechlen == len)
164  sasl->prefmech |= mechbit;
165  else
166  result = CURLE_URL_MALFORMAT;
167  }
168 
169  return result;
170 }

233  * Check if we have enough auth data and capabilities to authenticate.

235 bool Curl_sasl_can_authenticate(struct SASL *sasl, struct connectdata *conn)
236 {
237  /* Have credentials been provided? */
238  if(conn->bits.user_passwd)
239  return TRUE;
240 
241  /* EXTERNAL can authenticate without a user name and/or password */
242  if(sasl->authmechs & sasl->prefmech & SASL_MECH_EXTERNAL)
243  return TRUE;
244 
245  return FALSE;

248 /*
249  * Curl_sasl_start()
250  *
251  * Calculate the required login details for SASL authentication.
252  */
253 CURLcode Curl_sasl_start(struct SASL *sasl, struct connectdata *conn,
254  bool force_ir, saslprogress *progress)
255 {
256  CURLcode result = CURLE_OK;
257  struct Curl_easy *data = conn->data;
258  unsigned int enabledmechs;
259  const char *mech = NULL;
260  char *resp = NULL;
261  size_t len = 0;
262  saslstate state1 = SASL_STOP;
263  saslstate state2 = SASL_FINAL;
264  const char * const hostname = SSL_IS_PROXY() ? conn->http_proxy.host.name :
265  conn->host.name;
266  const long int port = SSL_IS_PROXY() ? conn->port : conn->remote_port;
267 #if defined(USE_KERBEROS5) || defined(USE_NTLM)
268  const char *service = data->set.str[STRING_SERVICE_NAME] ?
269  data->set.str[STRING_SERVICE_NAME] :
270  sasl->params->service;
271 #endif
272 
273  sasl->force_ir = force_ir; /* Latch for future use */
274  sasl->authused = 0; /* No mechanism used yet */
275  enabledmechs = sasl->authmechs & sasl->prefmech;
276  *progress = SASL_IDLE;
277 
278  /* Calculate the supported authentication mechanism, by decreasing order of
279  security, as well as the initial response where appropriate */
280  if((enabledmechs & SASL_MECH_EXTERNAL) && !conn->passwd[0]) {
281  mech = SASL_MECH_STRING_EXTERNAL;
282  state1 = SASL_EXTERNAL;
283  sasl->authused = SASL_MECH_EXTERNAL;
284 
285  if(force_ir || data->set.sasl_ir)
286  result = Curl_auth_create_external_message(data, conn->user, &resp,
287  &len);
288  }
289  else if(conn->bits.user_passwd) {
290 #if defined(USE_KERBEROS5)
291  if((enabledmechs & SASL_MECH_GSSAPI) && Curl_auth_is_gssapi_supported() &&
292  Curl_auth_user_contains_domain(conn->user)) {
293  sasl->mutual_auth = FALSE; /* TODO: Calculate mutual authentication */
294  mech = SASL_MECH_STRING_GSSAPI;
295  state1 = SASL_GSSAPI;
296  state2 = SASL_GSSAPI_TOKEN;
297  sasl->authused = SASL_MECH_GSSAPI;
298 
299  if(force_ir || data->set.sasl_ir)
300  result = Curl_auth_create_gssapi_user_message(data, conn->user,
301  conn->passwd,
302  service,
303  data->easy_conn->
304  host.name,
305  sasl->mutual_auth,
306  NULL, &conn->krb5,
307  &resp, &len);
308  }
309  else
310 #endif
311 #ifndef CURL_DISABLE_CRYPTO_AUTH
312  if((enabledmechs & SASL_MECH_DIGEST_MD5) &&
313  Curl_auth_is_digest_supported()) {
314  mech = SASL_MECH_STRING_DIGEST_MD5;
315  state1 = SASL_DIGESTMD5;
316  sasl->authused = SASL_MECH_DIGEST_MD5;
317  }
318  else if(enabledmechs & SASL_MECH_CRAM_MD5) {
319  mech = SASL_MECH_STRING_CRAM_MD5;
320  state1 = SASL_CRAMMD5;
321  sasl->authused = SASL_MECH_CRAM_MD5;
322  }
323  else
324 #endif
325 #ifdef USE_NTLM
326  if((enabledmechs & SASL_MECH_NTLM) && Curl_auth_is_ntlm_supported()) {
327  mech = SASL_MECH_STRING_NTLM;
328  state1 = SASL_NTLM;
329  state2 = SASL_NTLM_TYPE2MSG;
330  sasl->authused = SASL_MECH_NTLM;
331 
332  if(force_ir || data->set.sasl_ir)
333  result = Curl_auth_create_ntlm_type1_message(data,
334  conn->user, conn->passwd,
335  service,
336  hostname,
337  &conn->ntlm, &resp,
338  &len);
339  }
340  else
341 #endif
342  if((enabledmechs & SASL_MECH_OAUTHBEARER) && conn->oauth_bearer) {
343  mech = SASL_MECH_STRING_OAUTHBEARER;
344  state1 = SASL_OAUTH2;
345  state2 = SASL_OAUTH2_RESP;
346  sasl->authused = SASL_MECH_OAUTHBEARER;
347 
348  if(force_ir || data->set.sasl_ir)
349  result = Curl_auth_create_oauth_bearer_message(data, conn->user,
350  hostname,
351  port,
352  conn->oauth_bearer,
353  &resp, &len);
354  }
355  else if((enabledmechs & SASL_MECH_XOAUTH2) && conn->oauth_bearer) {
356  mech = SASL_MECH_STRING_XOAUTH2;
357  state1 = SASL_OAUTH2;
358  sasl->authused = SASL_MECH_XOAUTH2;
359 
360  if(force_ir || data->set.sasl_ir)
361  result = Curl_auth_create_oauth_bearer_message(data, conn->user,
362  NULL, 0,
363  conn->oauth_bearer,
364  &resp, &len);
365  }
366  else if(enabledmechs & SASL_MECH_PLAIN) {
367  mech = SASL_MECH_STRING_PLAIN;
368  state1 = SASL_PLAIN;
369  sasl->authused = SASL_MECH_PLAIN;
370 
371  if(force_ir || data->set.sasl_ir)
372  result = Curl_auth_create_plain_message(data, conn->user, conn->passwd,
373  &resp, &len);
374  }
375  else if(enabledmechs & SASL_MECH_LOGIN) {
376  mech = SASL_MECH_STRING_LOGIN;
377  state1 = SASL_LOGIN;
378  state2 = SASL_LOGIN_PASSWD;
379  sasl->authused = SASL_MECH_LOGIN;
380 
381  if(force_ir || data->set.sasl_ir)
382  result = Curl_auth_create_login_message(data, conn->user, &resp, &len);
383  }
384  }
385 
386  if(!result && mech) {
387  if(resp && sasl->params->maxirlen &&
388  strlen(mech) + len > sasl->params->maxirlen) {
389  free(resp);
390  resp = NULL;
391  }
392 
393  result = sasl->params->sendauth(conn, mech, resp);
394  if(!result) {
395  *progress = SASL_INPROGRESS;
396  state(sasl, conn, resp ? state2 : state1);
397  }
398  }
399 
400  free(resp);
401 
402  return result;
403 }

406  * Curl_sasl_continue()
407  *
408  * Continue the authentication.
409  */
410 CURLcode Curl_sasl_continue(struct SASL *sasl, struct connectdata *conn,
411  int code, saslprogress *progress)
412 {
413  CURLcode result = CURLE_OK;
414  struct Curl_easy *data = conn->data;
415  saslstate newstate = SASL_FINAL;
416  char *resp = NULL;
417  const char * const hostname = SSL_IS_PROXY() ? conn->http_proxy.host.name :
418  conn->host.name;
419  const long int port = SSL_IS_PROXY() ? conn->port : conn->remote_port;
420 #if !defined(CURL_DISABLE_CRYPTO_AUTH)
421  char *chlg = NULL;
422  size_t chlglen = 0;
423 #endif
424 #if !defined(CURL_DISABLE_CRYPTO_AUTH) || defined(USE_KERBEROS5) || \
425  defined(USE_NTLM)
426  const char *service = data->set.str[STRING_SERVICE_NAME] ?
427  data->set.str[STRING_SERVICE_NAME] :
428  sasl->params->service;
429  char *serverdata;
430 #endif
431  size_t len = 0;
432 
433  *progress = SASL_INPROGRESS;
434 
435  if(sasl->state == SASL_FINAL) {
436  if(code != sasl->params->finalcode)
437  result = CURLE_LOGIN_DENIED;
438  *progress = SASL_DONE;
439  state(sasl, conn, SASL_STOP);
440  return result;
441  }
442 
443  if(sasl->state != SASL_CANCEL && sasl->state != SASL_OAUTH2_RESP &&
444  code != sasl->params->contcode) {
445  *progress = SASL_DONE;
446  state(sasl, conn, SASL_STOP);
447  return CURLE_LOGIN_DENIED;
448  }
449 
450  switch(sasl->state) {
451  case SASL_STOP:
452  *progress = SASL_DONE;
453  return result;
454  case SASL_PLAIN:
455  result = Curl_auth_create_plain_message(data, conn->user, conn->passwd,
456  &resp,
457  &len);
458  break;
459  case SASL_LOGIN:
460  result = Curl_auth_create_login_message(data, conn->user, &resp, &len);
461  newstate = SASL_LOGIN_PASSWD;
462  break;
463  case SASL_LOGIN_PASSWD:
464  result = Curl_auth_create_login_message(data, conn->passwd, &resp, &len);
465  break;
466  case SASL_EXTERNAL:
467  result = Curl_auth_create_external_message(data, conn->user, &resp, &len);
468  break;
469 
470 #ifndef CURL_DISABLE_CRYPTO_AUTH
471  case SASL_CRAMMD5:
472  sasl->params->getmessage(data->state.buffer, &serverdata);
473  result = Curl_auth_decode_cram_md5_message(serverdata, &chlg, &chlglen);
474  if(!result)
475  result = Curl_auth_create_cram_md5_message(data, chlg, conn->user,
476  conn->passwd, &resp, &len);
477  free(chlg);
478  break;
479  case SASL_DIGESTMD5:
480  sasl->params->getmessage(data->state.buffer, &serverdata);
481  result = Curl_auth_create_digest_md5_message(data, serverdata,
482  conn->user, conn->passwd,
483  service,
484  &resp, &len);
485  newstate = SASL_DIGESTMD5_RESP;
486  break;
487  case SASL_DIGESTMD5_RESP:
488  resp = strdup("");
489  if(!resp)
490  result = CURLE_OUT_OF_MEMORY;
491  break;
492 #endif
493 
494 #ifdef USE_NTLM
495  case SASL_NTLM:
496  /* Create the type-1 message */
497  result = Curl_auth_create_ntlm_type1_message(data,
498  conn->user, conn->passwd,
499  service, hostname,
500  &conn->ntlm, &resp, &len);
501  newstate = SASL_NTLM_TYPE2MSG;
502  break;
503  case SASL_NTLM_TYPE2MSG:
504  /* Decode the type-2 message */
505  sasl->params->getmessage(data->state.buffer, &serverdata);
506  result = Curl_auth_decode_ntlm_type2_message(data, serverdata,
507  &conn->ntlm);
508  if(!result)
509  result = Curl_auth_create_ntlm_type3_message(data, conn->user,
510  conn->passwd, &conn->ntlm,
511  &resp, &len);
512  break;
513 #endif
514 
515 #if defined(USE_KERBEROS5)
516  case SASL_GSSAPI:
517  result = Curl_auth_create_gssapi_user_message(data, conn->user,
518  conn->passwd,
519  service,
520  data->easy_conn->host.name,
521  sasl->mutual_auth, NULL,
522  &conn->krb5,
523  &resp, &len);
524  newstate = SASL_GSSAPI_TOKEN;
525  break;
526  case SASL_GSSAPI_TOKEN:
527  sasl->params->getmessage(data->state.buffer, &serverdata);
528  if(sasl->mutual_auth) {
529  /* Decode the user token challenge and create the optional response
530  message */
531  result = Curl_auth_create_gssapi_user_message(data, NULL, NULL,
532  NULL, NULL,
533  sasl->mutual_auth,
534  serverdata, &conn->krb5,
535  &resp, &len);
536  newstate = SASL_GSSAPI_NO_DATA;
537  }
538  else
539  /* Decode the security challenge and create the response message */
540  result = Curl_auth_create_gssapi_security_message(data, serverdata,
541  &conn->krb5,
542  &resp, &len);
543  break;
544  case SASL_GSSAPI_NO_DATA:
545  sasl->params->getmessage(data->state.buffer, &serverdata);
546  /* Decode the security challenge and create the response message */
547  result = Curl_auth_create_gssapi_security_message(data, serverdata,
548  &conn->krb5,
549  &resp, &len);
550  break;
551 #endif
552 
553  case SASL_OAUTH2:
554  /* Create the authorisation message */
555  if(sasl->authused == SASL_MECH_OAUTHBEARER) {
556  result = Curl_auth_create_oauth_bearer_message(data, conn->user,
557  hostname,
558  port,
559  conn->oauth_bearer,
560  &resp, &len);
561 
562  /* Failures maybe sent by the server as continuations for OAUTHBEARER */
563  newstate = SASL_OAUTH2_RESP;
564  }
565  else
566  result = Curl_auth_create_oauth_bearer_message(data, conn->user,
567  NULL, 0,
568  conn->oauth_bearer,
569  &resp, &len);
570  break;
571 
572  case SASL_OAUTH2_RESP:
573  /* The continuation is optional so check the response code */
574  if(code == sasl->params->finalcode) {
575  /* Final response was received so we are done */
576  *progress = SASL_DONE;
577  state(sasl, conn, SASL_STOP);
578  return result;
579  }
580  else if(code == sasl->params->contcode) {
581  /* Acknowledge the continuation by sending a 0x01 response base64
582  encoded */
583  resp = strdup("AQ==");
584  if(!resp)
585  result = CURLE_OUT_OF_MEMORY;
586  break;
587  }
588  else {
589  *progress = SASL_DONE;
590  state(sasl, conn, SASL_STOP);
591  return CURLE_LOGIN_DENIED;
592  }
593 
594  case SASL_CANCEL:
595  /* Remove the offending mechanism from the supported list */
596  sasl->authmechs ^= sasl->authused;
597 
598  /* Start an alternative SASL authentication */
599  result = Curl_sasl_start(sasl, conn, sasl->force_ir, progress);
600  newstate = sasl->state; /* Use state from Curl_sasl_start() */
601  break;
602  default:
603  failf(data, "Unsupported SASL authentication mechanism");
604  result = CURLE_UNSUPPORTED_PROTOCOL; /* Should not happen */
605  break;
606  }
607 
608  switch(result) {
609  case CURLE_BAD_CONTENT_ENCODING:
610  /* Cancel dialog */
611  result = sasl->params->sendcont(conn, "*");
612  newstate = SASL_CANCEL;
613  break;
614  case CURLE_OK:
615  if(resp)
616  result = sasl->params->sendcont(conn, resp);
617  break;
618  default:
619  newstate = SASL_STOP; /* Stop on error */
620  *progress = SASL_DONE;
621  break;
622  }
623 
624  free(resp);
625 
626  state(sasl, conn, newstate);
627 
628  return result;
629 }
