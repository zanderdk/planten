109 void Adafruit_PCD8544::drawPixel(int16_t x, int16_t y, uint16_t color) {
110  if ((x < 0) || (x >= _width) || (y < 0) || (y >= _height)) {
111  return;
112  }
113 
114  int16_t t;
115  switch(rotation) {
116  case 1:
117  t = x;
118  x = y;
119  y = LCDHEIGHT - 1 - t;
120  break;
121 
122  case 2:
123  x = LCDWIDTH - 1 - x;
124  y = LCDHEIGHT - 1 - y;
125  break;
126 
127  case 3:
128  t = x;
129  x = LCDWIDTH - 1 - y;
130  y = t;
131  break;
132  }
133 
134  if ((x < 0) || (x >= LCDWIDTH) || (y < 0) || (y >= LCDHEIGHT))
135  return;
136 
137  // x is which column
138  if (color) {
139  pcd8544_buffer[x+ (y/8)*LCDWIDTH] |= _BV(y%8);
140  }
141  else {
142  pcd8544_buffer[x+ (y/8)*LCDWIDTH] &= ~_BV(y%8);
143  }
144 
145  updateBoundingBox(x,y,x,y);
146 }

265 void Adafruit_PCD8544::display(void) {
266  uint8_t col, maxcol, p;
267 
268  for(p = 0; p < 6; p++) {
269 #if defined(enablePartialUpdate)
270  // check if this page is part of update
271  if ( yUpdateMin >= ((p+1)*8) ) {
272  continue; // nope, skip it!
273  }
274  if (yUpdateMax < p*8) {
275  break;
276  }
277 #endif // enablePartialUpdate
278 
279  command(PCD8544_SETYADDR | p);
280 
281 
282 #if defined(enablePartialUpdate)
283  col = xUpdateMin;
284  maxcol = xUpdateMax;
285 #else // enablePartialUpdate
286  // start at the beginning of the row
287  col = 0;
288  maxcol = LCDWIDTH-1;
289 #endif // enablePartialUpdate
290 
291  command(PCD8544_SETXADDR | col);
292 
293  gpio_set_level((gpio_num_t)_dc, 1);
294 
295  for(; col <= maxcol; col++) {
296  spiWrite(pcd8544_buffer[(LCDWIDTH*p)+col]);
297  }
298  }
299 
300  command(PCD8544_SETYADDR ); // no idea why this is necessary but it is to finish the last byte?
301 #ifdef enablePartialUpdate
302  xUpdateMin = LCDWIDTH - 1;
303  xUpdateMax = 0;
304  yUpdateMin = LCDHEIGHT-1;
305  yUpdateMax = 0;
306 #endif
307 
308 }
