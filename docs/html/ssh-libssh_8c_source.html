300 /* Multiple options:
301  * 1. data->set.str[STRING_SSH_HOST_PUBLIC_KEY_MD5] is set with an MD5
302  * hash (90s style auth, not sure we should have it here)
303  * 2. data->set.ssh_keyfunc callback is set. Then we do trust on first
304  * use. We even save on knownhosts if CURLKHSTAT_FINE_ADD_TO_FILE

306  * 3. none of the above. We only accept if it is present on known hosts.

310 static int myssh_is_known(struct connectdata *conn)
311 {
312  int rc;
313  struct Curl_easy *data = conn->data;
314  struct ssh_conn *sshc = &conn->proto.sshc;
315  ssh_key pubkey;
316  size_t hlen;
317  unsigned char *hash = NULL;
318  char *base64 = NULL;
319  int vstate;
320  enum curl_khmatch keymatch;
321  struct curl_khkey foundkey;
322  curl_sshkeycallback func =
323  data->set.ssh_keyfunc;
324 
325  rc = ssh_get_publickey(sshc->ssh_session, &pubkey);
326  if(rc != SSH_OK)
327  return rc;
328 
329  if(data->set.str[STRING_SSH_HOST_PUBLIC_KEY_MD5]) {
330  rc = ssh_get_publickey_hash(pubkey, SSH_PUBLICKEY_HASH_MD5,
331  &hash, &hlen);
332  if(rc != SSH_OK)
333  goto cleanup;
334 
335  if(hlen != strlen(data->set.str[STRING_SSH_HOST_PUBLIC_KEY_MD5]) ||
336  memcmp(&data->set.str[STRING_SSH_HOST_PUBLIC_KEY_MD5], hash, hlen)) {
337  rc = SSH_ERROR;
338  goto cleanup;
339  }
340 
341  rc = SSH_OK;
342  goto cleanup;
343  }
344 
345  if(data->set.ssl.primary.verifyhost != TRUE) {
346  rc = SSH_OK;
347  goto cleanup;
348  }
349 
350  vstate = ssh_is_server_known(sshc->ssh_session);
351  switch(vstate) {
352  case SSH_SERVER_KNOWN_OK:
353  keymatch = CURLKHMATCH_OK;
354  break;
355  case SSH_SERVER_FILE_NOT_FOUND:
356  /* fallthrough */
357  case SSH_SERVER_NOT_KNOWN:
358  keymatch = CURLKHMATCH_MISSING;
359  break;
360  default:
361  keymatch = CURLKHMATCH_MISMATCH;
362  break;
363  }
364 
365  if(func) { /* use callback to determine action */
366  rc = ssh_pki_export_pubkey_base64(pubkey, &base64);
367  if(rc != SSH_OK)
368  goto cleanup;
369 
370  foundkey.key = base64;
371  foundkey.len = strlen(base64);
372 
373  switch(ssh_key_type(pubkey)) {
374  case SSH_KEYTYPE_RSA:
375  foundkey.keytype = CURLKHTYPE_RSA;
376  break;
377  case SSH_KEYTYPE_RSA1:
378  foundkey.keytype = CURLKHTYPE_RSA1;
379  break;
380  case SSH_KEYTYPE_ECDSA:
381  foundkey.keytype = CURLKHTYPE_ECDSA;
382  break;
383 #if LIBSSH_VERSION_INT >= SSH_VERSION_INT(0,7,0)
384  case SSH_KEYTYPE_ED25519:
385  foundkey.keytype = CURLKHTYPE_ED25519;
386  break;
387 #endif
388  case SSH_KEYTYPE_DSS:
389  foundkey.keytype = CURLKHTYPE_DSS;
390  break;
391  default:
392  rc = SSH_ERROR;
393  goto cleanup;
394  }
395 
396  /* we don't have anything equivalent to knownkey. Always NULL */
397  Curl_set_in_callback(data, true);
398  rc = func(data, NULL, &foundkey, /* from the remote host */
399  keymatch, data->set.ssh_keyfunc_userp);
400  Curl_set_in_callback(data, false);
401 
402  switch(rc) {
403  case CURLKHSTAT_FINE_ADD_TO_FILE:
404  rc = ssh_write_knownhost(sshc->ssh_session);
405  if(rc != SSH_OK) {
406  goto cleanup;
407  }
408  break;
409  case CURLKHSTAT_FINE:
410  break;
411  default: /* REJECT/DEFER */
412  rc = SSH_ERROR;
413  goto cleanup;
414  }
415  }
416  else {
417  if(keymatch != CURLKHMATCH_OK) {
418  rc = SSH_ERROR;
419  goto cleanup;
420  }
421  }
422  rc = SSH_OK;
423 
424 cleanup:
425  if(hash)
426  ssh_clean_pubkey_hash(&hash);
427  ssh_key_free(pubkey);
428  return rc;
429 }

482 restart:

484  case 0:
485  rc = ssh_userauth_kbdint(sshc->ssh_session, NULL, NULL);
486  if(rc == SSH_AUTH_AGAIN)
487  return SSH_AGAIN;
488 
489  if(rc != SSH_AUTH_INFO)

491

495

497  if(rc < 0)
498  return SSH_ERROR;
499 
500  /* FALLTHROUGH */
501  case 1:
502  sshc->kbd_state = 1;
503 
504  rc = ssh_userauth_kbdint(sshc->ssh_session, NULL, NULL);
505  if(rc == SSH_AUTH_AGAIN)
506  return SSH_AGAIN;
507  else if(rc == SSH_AUTH_SUCCESS)
508  rc = SSH_OK;
509  else if(rc == SSH_AUTH_INFO) {
510  nprompts = ssh_userauth_kbdint_getnprompts(sshc->ssh_session);
511  if(nprompts != 0)
512  return SSH_ERROR;
513 
514  sshc->kbd_state = 2;
515  goto restart;
516  }
517  else
518  rc = SSH_ERROR;
519  break;
520  case 2:
521  sshc->kbd_state = 2;
522 
523  rc = ssh_userauth_kbdint(sshc->ssh_session, NULL, NULL);
524  if(rc == SSH_AUTH_AGAIN)
525  return SSH_AGAIN;
526  else if(rc == SSH_AUTH_SUCCESS)
527  rc = SSH_OK;
528  else
529  rc = SSH_ERROR;
530 
531  break;
532  default:
533  return SSH_ERROR;
534  }
535 
536  sshc->kbd_state = 0;

540 /*
541  * ssh_statemach_act() runs the SSH state machine as far as it can without
542  * blocking and without reaching the end. The data the pointer 'block' points
543  * to will be set to TRUE if the libssh function returns SSH_AGAIN
544  * meaning it wants to be called again when the socket is ready
545  */
546 static CURLcode myssh_statemach_act(struct connectdata *conn, bool *block)
547 {
548  CURLcode result = CURLE_OK;
549  struct Curl_easy *data = conn->data;
550  struct SSHPROTO *protop = data->req.protop;
551  struct ssh_conn *sshc = &conn->proto.sshc;
552  int rc = SSH_NO_ERROR, err;
553  char *new_readdir_line;
554  int seekerr = CURL_SEEKFUNC_OK;
555  const char *err_msg;
556  *block = 0; /* we're not blocking by default */
557 
558  do {
559 
560  switch(sshc->state) {
561  case SSH_INIT:
562  sshc->secondCreateDirs = 0;
563  sshc->nextstate = SSH_NO_STATE;
564  sshc->actualcode = CURLE_OK;
565 
566 #if 0
567  ssh_set_log_level(SSH_LOG_PROTOCOL);
568 #endif
569 
570  /* Set libssh to non-blocking, since everything internally is
571  non-blocking */
572  ssh_set_blocking(sshc->ssh_session, 0);
573 
574  state(conn, SSH_S_STARTUP);
575  /* FALLTHROUGH */
576 
577  case SSH_S_STARTUP:
578  rc = ssh_connect(sshc->ssh_session);
579  if(rc == SSH_AGAIN)
580  break;
581 
582  if(rc != SSH_OK) {
583  failf(data, "Failure establishing ssh session");
584  MOVE_TO_ERROR_STATE(CURLE_FAILED_INIT);
585  }
586 
587  state(conn, SSH_HOSTKEY);
588 
589  /* FALLTHROUGH */
590  case SSH_HOSTKEY:
591 
592  rc = myssh_is_known(conn);
593  if(rc != SSH_OK) {
594  MOVE_TO_ERROR_STATE(CURLE_PEER_FAILED_VERIFICATION);
595  }
596 
597  state(conn, SSH_AUTHLIST);
598  /* FALLTHROUGH */
599  case SSH_AUTHLIST:{
600  sshc->authed = FALSE;
601 
602  rc = ssh_userauth_none(sshc->ssh_session, NULL);
603  if(rc == SSH_AUTH_AGAIN) {
604  rc = SSH_AGAIN;
605  break;
606  }
607 
608  if(rc == SSH_AUTH_SUCCESS) {
609  sshc->authed = TRUE;
610  infof(data, "Authenticated with none\n");
611  state(conn, SSH_AUTH_DONE);
612  break;
613  }
614  else if(rc == SSH_AUTH_ERROR) {
615  MOVE_TO_ERROR_STATE(CURLE_LOGIN_DENIED);
616  }
617 
618  sshc->auth_methods = ssh_userauth_list(sshc->ssh_session, NULL);
619  if(sshc->auth_methods & SSH_AUTH_METHOD_PUBLICKEY) {
620  state(conn, SSH_AUTH_PKEY_INIT);
621  infof(data, "Authentication using SSH public key file\n");
622  }
623  else if(sshc->auth_methods & SSH_AUTH_METHOD_GSSAPI_MIC) {
624  state(conn, SSH_AUTH_GSSAPI);
625  }
626  else if(sshc->auth_methods & SSH_AUTH_METHOD_INTERACTIVE) {
627  state(conn, SSH_AUTH_KEY_INIT);
628  }
629  else if(sshc->auth_methods & SSH_AUTH_METHOD_PASSWORD) {
630  state(conn, SSH_AUTH_PASS_INIT);
631  }
632  else { /* unsupported authentication method */
633  MOVE_TO_ERROR_STATE(CURLE_LOGIN_DENIED);
634  }
635 
636  break;
637  }
638  case SSH_AUTH_PKEY_INIT:
639  if(!(data->set.ssh_auth_types & CURLSSH_AUTH_PUBLICKEY)) {
640  MOVE_TO_SECONDARY_AUTH;
641  }
642 
643  /* Two choices, (1) private key was given on CMD,
644  * (2) use the "default" keys. */
645  if(data->set.str[STRING_SSH_PRIVATE_KEY]) {
646  if(sshc->pubkey && !data->set.ssl.key_passwd) {
647  rc = ssh_userauth_try_publickey(sshc->ssh_session, NULL,
648  sshc->pubkey);
649  if(rc == SSH_AUTH_AGAIN) {
650  rc = SSH_AGAIN;
651  break;
652  }
653 
654  if(rc != SSH_OK) {
655  MOVE_TO_SECONDARY_AUTH;
656  }
657  }
658 
659  rc = ssh_pki_import_privkey_file(data->
660  set.str[STRING_SSH_PRIVATE_KEY],
661  data->set.ssl.key_passwd, NULL,
662  NULL, &sshc->privkey);
663  if(rc != SSH_OK) {
664  failf(data, "Could not load private key file %s",
665  data->set.str[STRING_SSH_PRIVATE_KEY]);
666  MOVE_TO_ERROR_STATE(CURLE_LOGIN_DENIED);
667  break;
668  }
669 
670  state(conn, SSH_AUTH_PKEY);
671  break;
672 
673  }
674  else {
675  rc = ssh_userauth_publickey_auto(sshc->ssh_session, NULL,
676  data->set.ssl.key_passwd);

754  break;

770

856  sshc->quote_item = data->set.quote;

859  else {
860  state(conn, SSH_SFTP_GETINFO);
861  }
862  break;
863 
864  case SSH_SFTP_POSTQUOTE_INIT:

866  infof(data, "Sending quote commands\n");

870  else {
871  state(conn, SSH_STOP);
872  }

916  * we do not send the actual error; we return

933  sshc->actualcode = CURLE_QUOTE_ERROR;

935  }

948  sshc->actualcode = CURLE_QUOTE_ERROR;

951  state(conn, SSH_SFTP_NEXT_QUOTE);
952  break;

959  Curl_safefree(sshc->quote_path2);

961  ssh_get_error(sshc->ssh_session));

964  sshc->actualcode = CURLE_QUOTE_ERROR;
965  break;
966  }
967  state(conn, SSH_SFTP_NEXT_QUOTE);
968  break;

972  if(rc != 0 && !sshc->acceptfail) {
973  Curl_safefree(sshc->quote_path1);
974  failf(data, "rmdir command failed: %s",
975  ssh_get_error(sshc->ssh_session));
976  state(conn, SSH_SFTP_CLOSE);
977  sshc->nextstate = SSH_NO_STATE;
978  sshc->actualcode = CURLE_QUOTE_ERROR;
979  break;
980  }
981  state(conn, SSH_SFTP_NEXT_QUOTE);
982  break;
983

985  rc = sftp_unlink(sshc->sftp_session, sshc->quote_path1);
986  if(rc != 0 && !sshc->acceptfail) {
987  Curl_safefree(sshc->quote_path1);
988  failf(data, "rm command failed: %s",
989  ssh_get_error(sshc->ssh_session));
990  state(conn, SSH_SFTP_CLOSE);
991  sshc->nextstate = SSH_NO_STATE;
992  sshc->actualcode = CURLE_QUOTE_ERROR;
993  break;
994  }
995  state(conn, SSH_SFTP_NEXT_QUOTE);
996  break;
997 
998  case SSH_SFTP_QUOTE_STATVFS:
999  {
1000  sftp_statvfs_t statvfs;
1001 
1002  statvfs = sftp_statvfs(sshc->sftp_session, sshc->quote_path1);
1003  if(!statvfs && !sshc->acceptfail) {
1004  Curl_safefree(sshc->quote_path1);
1005  failf(data, "statvfs command failed: %s",
1006  ssh_get_error(sshc->ssh_session));
1007  state(conn, SSH_SFTP_CLOSE);
1008  sshc->nextstate = SSH_NO_STATE;
1009  sshc->actualcode = CURLE_QUOTE_ERROR;
1010  break;
1011  }
1012  else if(statvfs) {
1013  char *tmp = aprintf("statvfs:\n"
1014  "f_bsize: %llu\n" "f_frsize: %llu\n"
1015  "f_blocks: %llu\n" "f_bfree: %llu\n"
1016  "f_bavail: %llu\n" "f_files: %llu\n"

1019  "f_namemax: %llu\n",
1020  statvfs->f_bsize, statvfs->f_frsize,
1021  statvfs->f_blocks, statvfs->f_bfree,
1022  statvfs->f_bavail, statvfs->f_files,
1023  statvfs->f_ffree, statvfs->f_favail,
1024  statvfs->f_fsid, statvfs->f_flag,
1025  statvfs->f_namemax);

1027

1029  result = CURLE_OUT_OF_MEMORY;
1030  state(conn, SSH_SFTP_CLOSE);
1031  sshc->nextstate = SSH_NO_STATE;
1032  break;
1033  }
1034

1036  free(tmp);
1037  if(result) {
1038  state(conn, SSH_SFTP_CLOSE);
1039  sshc->nextstate = SSH_NO_STATE;
1040  sshc->actualcode = result;
1041  }
1042  }
1043  state(conn, SSH_SFTP_NEXT_QUOTE);
1044  break;
1045  }
1046 
1047  case SSH_SFTP_GETINFO:
1048  if(data->set.get_filetime) {
1049  state(conn, SSH_SFTP_FILETIME);
1050  }
1051  else {
1052  state(conn, SSH_SFTP_TRANS_INIT);
1053  }
1054  break;
1055 
1056  case SSH_SFTP_FILETIME:
1057  {
1058  sftp_attributes attrs;
1059 
1060  attrs = sftp_stat(sshc->sftp_session, protop->path);
1061  if(attrs != 0) {
1062  data->info.filetime = attrs->mtime;
1063  sftp_attributes_free(attrs);
1064  }
1065 
1066  state(conn, SSH_SFTP_TRANS_INIT);
1067  break;
1068  }
1069 
1070  case SSH_SFTP_TRANS_INIT:
1071  if(data->set.upload)
1072  state(conn, SSH_SFTP_UPLOAD_INIT);
1073  else {
1074  if(protop->path[strlen(protop->path)-1] == '/')
1075  state(conn, SSH_SFTP_READDIR_INIT);
1076  else
1077  state(conn, SSH_SFTP_DOWNLOAD_INIT);
1078  }
1079  break;
1080 
1081  case SSH_SFTP_UPLOAD_INIT:
1082  {
1083  int flags;
1084 
1085  if(data->state.resume_from != 0) {
1086  sftp_attributes attrs;
1087 
1088  if(data->state.resume_from < 0) {
1089  attrs = sftp_stat(sshc->sftp_session, protop->path);
1090  if(attrs != 0) {
1091  curl_off_t size = attrs->size;
1092  if(size < 0) {
1093  failf(data, "Bad file size (%" CURL_FORMAT_CURL_OFF_T ")", size);
1094  MOVE_TO_ERROR_STATE(CURLE_BAD_DOWNLOAD_RESUME);
1095  }
1096  data->state.resume_from = attrs->size;
1097 
1098  sftp_attributes_free(attrs);
1099  }
1100  else {
1101  data->state.resume_from = 0;
1102  }
1103  }
1104  }
1105 
1106  if(data->set.ftp_append)
1107  /* Try to open for append, but create if nonexisting */
1108  flags = O_WRONLY|O_CREAT|O_APPEND;
1109  else if(data->state.resume_from > 0)
1110  /* If we have restart position then open for append */
1111  flags = O_WRONLY|O_APPEND;
1112  else
1113  /* Clear file before writing (normal behaviour) */
1114  flags = O_WRONLY|O_APPEND|O_CREAT|O_TRUNC;
1115 
1116  if(sshc->sftp_file)
1117  sftp_close(sshc->sftp_file);
1118  sshc->sftp_file =
1119  sftp_open(sshc->sftp_session, protop->path,
1120  flags, (mode_t)data->set.new_file_perms);
1121  if(!sshc->sftp_file) {
1122  err = sftp_get_error(sshc->sftp_session);
1123 
1124  if(((err == SSH_FX_NO_SUCH_FILE || err == SSH_FX_FAILURE ||
1125  err == SSH_FX_NO_SUCH_PATH)) &&
1126  (data->set.ftp_create_missing_dirs &&
1127  (strlen(protop->path) > 1))) {
1128  /* try to create the path remotely */
1129  rc = 0;
1130  sshc->secondCreateDirs = 1;
1131  state(conn, SSH_SFTP_CREATE_DIRS_INIT);
1132  break;
1133  }
1134  else {
1135  MOVE_TO_SFTP_CLOSE_STATE();
1136  }
1137  }
1138 
1139  /* If we have a restart point then we need to seek to the correct
1140  position. */
1141  if(data->state.resume_from > 0) {
1142  /* Let's read off the proper amount of bytes from the input. */
1143  if(conn->seek_func) {
1144  Curl_set_in_callback(data, true);
1145  seekerr = conn->seek_func(conn->seek_client, data->state.resume_from,
1146  SEEK_SET);
1147  Curl_set_in_callback(data, false);
1148  }
1149 
1150  if(seekerr != CURL_SEEKFUNC_OK) {
1151  curl_off_t passed = 0;
1152 
1153  if(seekerr != CURL_SEEKFUNC_CANTSEEK) {
1154  failf(data, "Could not seek stream");
1155  return CURLE_FTP_COULDNT_USE_REST;
1156  }
1157  /* seekerr == CURL_SEEKFUNC_CANTSEEK (can't seek to offset) */
1158  do {
1159  size_t readthisamountnow =
1160  (data->state.resume_from - passed > data->set.buffer_size) ?
1161  (size_t)data->set.buffer_size :
1162  curlx_sotouz(data->state.resume_from - passed);
1163 
1164  size_t actuallyread =
1165  data->state.fread_func(data->state.buffer, 1,
1166  readthisamountnow, data->state.in);
1167 
1168  passed += actuallyread;
1169  if((actuallyread == 0) || (actuallyread > readthisamountnow)) {
1170  /* this checks for greater-than only to make sure that the
1171  CURL_READFUNC_ABORT return code still aborts */
1172  failf(data, "Failed to read data");
1173  MOVE_TO_ERROR_STATE(CURLE_FTP_COULDNT_USE_REST);
1174  }
1175  } while(passed < data->state.resume_from);
1176  }
1177 
1178  /* now, decrease the size of the read */
1179  if(data->state.infilesize > 0) {
1180  data->state.infilesize -= data->state.resume_from;
1181  data->req.size = data->state.infilesize;
1182  Curl_pgrsSetUploadSize(data, data->state.infilesize);
1183  }
1184 
1185  rc = sftp_seek64(sshc->sftp_file, data->state.resume_from);
1186  if(rc != 0) {
1187  MOVE_TO_SFTP_CLOSE_STATE();
1188  }
1189  }
1190  if(data->state.infilesize > 0) {
1191  data->req.size = data->state.infilesize;
1192  Curl_pgrsSetUploadSize(data, data->state.infilesize);
1193  }
1194  /* upload data */
1195  Curl_setup_transfer(conn, -1, -1, FALSE, NULL, FIRSTSOCKET, NULL);
1196 
1197  /* not set by Curl_setup_transfer to preserve keepon bits */
1198  conn->sockfd = conn->writesockfd;
1199 
1200  /* store this original bitmask setup to use later on if we can't
1201  figure out a "real" bitmask */
1202  sshc->orig_waitfor = data->req.keepon;
1203 
1204  /* we want to use the _sending_ function even when the socket turns
1205  out readable as the underlying libssh sftp send function will deal
1206  with both accordingly */
1207  conn->cselect_bits = CURL_CSELECT_OUT;
1208 
1209  /* since we don't really wait for anything at this point, we want the
1210  state machine to move on as soon as possible so we set a very short
1211  timeout here */
1212  Curl_expire(data, 0, EXPIRE_RUN_NOW);
1213 
1214  state(conn, SSH_STOP);
1215  break;
1216  }
1217 
1218  case SSH_SFTP_CREATE_DIRS_INIT:
1219  if(strlen(protop->path) > 1) {
1220  sshc->slash_pos = protop->path + 1; /* ignore the leading '/' */
1221  state(conn, SSH_SFTP_CREATE_DIRS);
1222  }
1223  else {
1224  state(conn, SSH_SFTP_UPLOAD_INIT);
1225  }
1226  break;
1227 
1228  case SSH_SFTP_CREATE_DIRS:
1229  sshc->slash_pos = strchr(sshc->slash_pos, '/');
1230  if(sshc->slash_pos) {
1231  *sshc->slash_pos = 0;
1232 
1233  infof(data, "Creating directory '%s'\n", protop->path);
1234  state(conn, SSH_SFTP_CREATE_DIRS_MKDIR);
1235  break;
1236  }
1237  state(conn, SSH_SFTP_UPLOAD_INIT);
1238  break;
1239 
1240  case SSH_SFTP_CREATE_DIRS_MKDIR:
1241  /* 'mode' - parameter is preliminary - default to 0644 */
1242  rc = sftp_mkdir(sshc->sftp_session, protop->path,
1243  (mode_t)data->set.new_directory_perms);
1244  *sshc->slash_pos = '/';
1245  ++sshc->slash_pos;
1246  if(rc < 0) {
1247  /*
1248  * Abort if failure wasn't that the dir already exists or the
1249  * permission was denied (creation might succeed further down the
1250  * path) - retry on unspecific FAILURE also
1251  */
1252  err = sftp_get_error(sshc->sftp_session);
1253  if((err != SSH_FX_FILE_ALREADY_EXISTS) &&
1254  (err != SSH_FX_FAILURE) &&
1255  (err != SSH_FX_PERMISSION_DENIED)) {
1256  MOVE_TO_SFTP_CLOSE_STATE();
1257  }
1258  rc = 0; /* clear rc and continue */
1259  }
1260  state(conn, SSH_SFTP_CREATE_DIRS);
1261  break;
1262 
1263  case SSH_SFTP_READDIR_INIT:
1264  Curl_pgrsSetDownloadSize(data, -1);
1265  if(data->set.opt_no_body) {
1266  state(conn, SSH_STOP);
1267  break;
1268  }
1269 
1270  /*
1271  * This is a directory that we are trying to get, so produce a directory
1272  * listing
1273  */
1274  sshc->sftp_dir = sftp_opendir(sshc->sftp_session,
1275  protop->path);
1276  if(!sshc->sftp_dir) {
1277  failf(data, "Could not open directory for reading: %s",
1278  ssh_get_error(sshc->ssh_session));
1279  MOVE_TO_SFTP_CLOSE_STATE();
1280  }
1281  state(conn, SSH_SFTP_READDIR);
1282  break;
1283
