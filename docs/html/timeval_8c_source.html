111  /*
112  ** Monotonic timer on Mac OS is provided by mach_absolute_time(), which
113  ** returns time in Mach "absolute time units," which are platform-dependent.
114  ** To convert to nanoseconds, one must use conversion factors specified by
115  ** mach_timebase_info().
116  */
117  static mach_timebase_info_data_t timebase;
118  struct curltime cnow;
119  uint64_t usecs;
120 
121  if(0 == timebase.denom)
122  (void) mach_timebase_info(&timebase);
123 
124  usecs = mach_absolute_time();
125  usecs *= timebase.numer;
126  usecs /= timebase.denom;
127  usecs /= 1000;
128 
129  cnow.tv_sec = usecs / 1000000;
130  cnow.tv_usec = (int)(usecs % 1000000);
