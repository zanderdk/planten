200 void testdrawbitmap(const uint8_t *bitmap, uint8_t w, uint8_t h) {
201  uint8_t icons[NUMFLAKES][3];
202  //randomSeed(666); // whatever seed
203 
204  // initialize
205  for (uint8_t f=0; f< NUMFLAKES; f++) {
206  icons[f][XPOS] = rand() % display.width();
207  icons[f][YPOS] = 0;
208  icons[f][DELTAY] = rand()%5 + 1;
209 
210  /*
211  Serial.print("x: ");
212  Serial.print(icons[f][XPOS], DEC);
213  Serial.print(" y: ");
214  Serial.print(icons[f][YPOS], DEC);
215  Serial.print(" dy: ");
216  Serial.println(icons[f][DELTAY], DEC);
217  */
218  }
219 
220  while (1) {
221  // draw each icon
222  for (uint8_t f=0; f< NUMFLAKES; f++) {
223  display.drawBitmap(icons[f][XPOS], icons[f][YPOS], logo16_glcd_bmp, w, h, BLACK);
224  }
225  display.display();
226  delay(200);
227 
228  // then erase it + move it
229  for (uint8_t f=0; f< NUMFLAKES; f++) {
230  display.drawBitmap(icons[f][XPOS], icons[f][YPOS], logo16_glcd_bmp, w, h, WHITE);
231  // move it
232  icons[f][YPOS] += icons[f][DELTAY];
233  // if its gone, reinit
234  if (icons[f][YPOS] > display.height()) {
235  icons[f][XPOS] = rand() % display.width();
236  icons[f][YPOS] = 0;
237  icons[f][DELTAY] = rand()%5 + 1;
238  }
239  }
240  }
241 }
