66 static size_t write_callback(void *ptr, size_t size, size_t nmemb, void *userp)
67 {
68  struct transfer_status *st = (struct transfer_status *)userp;
69  size_t len = size * nmemb;
70  st->counter++;
71  if(st->counter > 1) {
