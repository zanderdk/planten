129 #ifdef __VMS
130 /*
131  * get_vms_file_size does what it takes to get the real size of the file
132  *
133  * For fixed files, find out the size of the EOF block and adjust.
134  *
135  * For all others, have to read the entire file in, discarding the contents.
136  * Most posted text files will be small, and binary files like zlib archives
137  * and CD/DVD images should be either a STREAM_LF format or a fixed format.
138  *
139  */
140 static curl_off_t vms_realfilesize(const char *name,
141  const struct_stat *stat_buf)
142 {
143  char buffer[8192];
144  curl_off_t count;
145  int ret_stat;
146  FILE * file;
147 
148  /* !checksrc! disable FOPENMODE 1 */
149  file = fopen(name, "r"); /* VMS */
150  if(file == NULL) {
151  return 0;
152  }
153  count = 0;
154  ret_stat = 1;
155  while(ret_stat > 0) {
156  ret_stat = fread(buffer, 1, sizeof(buffer), file);
157  if(ret_stat != 0)
158  count += ret_stat;
159  }
160  fclose(file);
161 
162  return count;
163 }
