94 static char *unescape_word(struct Curl_easy *data, const char *inputbuff)
95 {
96  char *newp = NULL;
97  char *dictp;
98  size_t len;
99 
100  CURLcode result = Curl_urldecode(data, inputbuff, 0, &newp, &len, FALSE);
101  if(!newp || result)
102  return NULL;
103 
104  dictp = malloc(len*2 + 1); /* add one for terminating zero */
105  if(dictp) {
106  char *ptr;
107  char ch;
108  int olen = 0;
109  /* According to RFC2229 section 2.2, these letters need to be escaped with
110  \[letter] */
111  for(ptr = newp;
112  (ch = *ptr) != 0;
113  ptr++) {
114  if((ch <= 32) || (ch == 127) ||
115  (ch == '\'') || (ch == '\"') || (ch == '\\')) {
116  dictp[olen++] = '\\';
117  }
118  dictp[olen++] = ch;
119  }
120  dictp[olen] = 0;
121  }
122  free(newp);
123  return dictp;
124 }
