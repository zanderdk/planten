66  m_pServer = pServer;

134 void BLEService::start() {
135 // We ask the BLE runtime to start the service and then create each of the characteristics.
136 // We start the service through its local handle which was returned in the ESP_GATTS_CREATE_EVT event
137 // obtained as a result of calling esp_ble_gatts_create_service().
138 //
139  ESP_LOGD(LOG_TAG, ">> start(): Starting service (esp_ble_gatts_start_service): %s", toString().c_str());
140  if (m_handle == NULL_HANDLE) {
141  ESP_LOGE(LOG_TAG, "<< !!! We attempted to start a service but don't know its handle!");
142  return;
143  }
144 
145  BLECharacteristic *pCharacteristic = m_characteristicMap.getFirst();
146 
147  while (pCharacteristic != nullptr) {
148  m_lastCreatedCharacteristic = pCharacteristic;
149  pCharacteristic->executeCreate(this);
150 
151  pCharacteristic = m_characteristicMap.getNext();
152  }
153  // Start each of the characteristics ... these are found in the m_characteristicMap.
154 
155  m_semaphoreStartEvt.take("start");
156  esp_err_t errRc = ::esp_ble_gatts_start_service(m_handle);
157 
158  if (errRc != ESP_OK) {
159  ESP_LOGE(LOG_TAG, "<< esp_ble_gatts_start_service: rc=%d %s", errRc, GeneralUtils::errorToString(errRc));
160  return;
161  }
162  m_semaphoreStartEvt.wait("start");
163 
164  ESP_LOGD(LOG_TAG, "<< start()");
165 } // start

222 void BLEService::addCharacteristic(BLECharacteristic* pCharacteristic) {
223  // We maintain a mapping of characteristics owned by this service. These are managed by the
224  // BLECharacteristicMap class instance found in m_characteristicMap. We add the characteristic
225  // to the map and then ask the service to add the characteristic at the BLE level (ESP-IDF).
226 
227  ESP_LOGD(LOG_TAG, ">> addCharacteristic()");
228  ESP_LOGD(LOG_TAG, "Adding characteristic: uuid=%s to service: %s",
229  pCharacteristic->getUUID().toString().c_str(),
230  toString().c_str());
231 
232  // Check that we don't add the same characteristic twice.
233  if (m_characteristicMap.getByUUID(pCharacteristic->getUUID()) != nullptr) {
234  ESP_LOGW(LOG_TAG, "<< Adding a new characteristic with the same UUID as a previous one");
235  //return;
236  }
237 
238  // Remember this characteristic in our map of characteristics. At this point, we can lookup by UUID
239  // but not by handle. The handle is allocated to us on the ESP_GATTS_ADD_CHAR_EVT.
240  m_characteristicMap.setByUUID(pCharacteristic, pCharacteristic->getUUID());
241 
242  ESP_LOGD(LOG_TAG, "<< addCharacteristic()");
243 } // addCharacteristic

273 void BLEService::handleGATTServerEvent(esp_gatts_cb_event_t event, esp_gatt_if_t gatts_if, esp_ble_gatts_cb_param_t* param) {
274  switch (event) {
275  // ESP_GATTS_ADD_CHAR_EVT - Indicate that a characteristic was added to the service.
276  // add_char:
277  // - esp_gatt_status_t status
278  // - uint16_t attr_handle
279  // - uint16_t service_handle
280  // - esp_bt_uuid_t char_uuid
281 
282  // If we have reached the correct service, then locate the characteristic and remember the handle
283  // for that characteristic.
284  case ESP_GATTS_ADD_CHAR_EVT: {
285  if (m_handle == param->add_char.service_handle) {
286  BLECharacteristic *pCharacteristic = getLastCreatedCharacteristic();
287  if (pCharacteristic == nullptr) {
288  ESP_LOGE(LOG_TAG, "Expected to find characteristic with UUID: %s, but didnt!",
289  BLEUUID(param->add_char.char_uuid).toString().c_str());
290  dump();
291  break;
292  }
293  pCharacteristic->setHandle(param->add_char.attr_handle);
294  m_characteristicMap.setByHandle(param->add_char.attr_handle, pCharacteristic);
295  break;
296  } // Reached the correct service.
297  break;
298  } // ESP_GATTS_ADD_CHAR_EVT
299 
300 
301  // ESP_GATTS_START_EVT
302  //
303  // start:
304  // esp_gatt_status_t status
305  // uint16_t service_handle
306  case ESP_GATTS_START_EVT: {
307  if (param->start.service_handle == getHandle()) {
308  m_semaphoreStartEvt.give();
309  }
310  break;
311  } // ESP_GATTS_START_EVT
312 
313  // ESP_GATTS_STOP_EVT
314  //
315  // stop:
316  // esp_gatt_status_t status
317  // uint16_t service_handle
318  //
319  case ESP_GATTS_STOP_EVT: {
320  if (param->stop.service_handle == getHandle()) {
321  m_semaphoreStopEvt.give();
322  }
323  break;
324  } // ESP_GATTS_STOP_EVT
325 
326 
327  // ESP_GATTS_CREATE_EVT
328  // Called when a new service is registered as having been created.
329  //
330  // create:
331  // * esp_gatt_status_t status
332  // * uint16_t service_handle
333  // * esp_gatt_srvc_id_t service_id
334  // * - esp_gatt_id id
335  // * - esp_bt_uuid uuid
336  // * - uint8_t inst_id
337  // * - bool is_primary
338  //
339  case ESP_GATTS_CREATE_EVT: {
340  if (getUUID().equals(BLEUUID(param->create.service_id.id.uuid)) && m_instId == param->create.service_id.id.inst_id) {
341  setHandle(param->create.service_handle);
342  m_semaphoreCreateEvt.give();
343  }
344  break;
345  } // ESP_GATTS_CREATE_EVT
346 
347 
348  // ESP_GATTS_DELETE_EVT
349  // Called when a service is deleted.
350  //
351  // delete:
352  // * esp_gatt_status_t status
353  // * uint16_t service_handle
354  //
355  case ESP_GATTS_DELETE_EVT: {
356  if (param->del.service_handle == getHandle()) {
357  m_semaphoreDeleteEvt.give();
358  }
359  break;
360  } // ESP_GATTS_DELETE_EVT
361 
362  default:
363  break;
364  } // Switch
365 
366  // Invoke the GATTS handler in each of the associated characteristics.
367  m_characteristicMap.handleGATTServerEvent(event, gatts_if, param);
368 } // handleGATTServerEvent
