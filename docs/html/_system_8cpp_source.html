403 /* static */ void System::dumpPinMapping() {
404  const int numPins = 40;
405  printf("GPIO_FUNCn_OUT_SEL_CFG_REG\n");
406  printf("--------------------------\n");
407  printf("%3s %4s\n", "Pin", "Func");
408  for (uint8_t i = 0; i < numPins; i++) {
409  const char *signal;
410  if (GPIO.func_out_sel_cfg[i].func_sel == 256) {
411  signal = (char*) "[GPIO]";
412  } else if (GPIO.func_out_sel_cfg[i].func_sel == 257) {
413  signal = (char*) "N/A";
414  } else {
415  signal = outSignalStrings[GPIO.func_out_sel_cfg[i].func_sel];
416  }
417  printf("%2d %4d %s\n", i, GPIO.func_out_sel_cfg[i].func_sel, signal);
418  const io_mux_reg_t* io_mux = gpioToIoMux(i);
419  if (GPIO.func_out_sel_cfg[i].func_sel == 256 && io_mux != nullptr) {
420  printf("0x%x - function: %d, ie: %d\n", (uint32_t) io_mux, io_mux->mcu_sel + 1, io_mux->func_ie);
421  }
422  }
