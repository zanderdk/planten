105 static int linux_read(Network* n, unsigned char* buffer, int len, int timeout_ms)
106 {
107  struct timeval interval = {timeout_ms / 1000, (timeout_ms % 1000) * 1000};
108  if (interval.tv_sec < 0 || (interval.tv_sec == 0 && interval.tv_usec <= 0))
109  {
110  interval.tv_sec = 0;
111  interval.tv_usec = 100;
112  }
113 
114  setsockopt(n->my_socket, SOL_SOCKET, SO_RCVTIMEO, (char *)&interval, sizeof(struct timeval));
115 
116  int bytes = 0;
117  while (bytes < len) {
118  int rc = recv(n->my_socket, &buffer[bytes], (size_t)(len - bytes), 0);
119  if (rc == -1) {
120  if (errno != ENOTCONN && errno != ECONNRESET) {
121  bytes = -1;
122  break;
123  }
124  }
125  else {
126  bytes += rc;
127  }
128  }
129  return bytes;
130 }

273 int NetworkConnect(Network* n, char* addr, int port)
274 {
275  if (n->isSSL) {
276  return NetworkConnectSSL(n, addr, port);
277  }
278  int type = SOCK_STREAM;
279  struct sockaddr_in address;
280  int rc = -1;
281  sa_family_t family = AF_INET;
282  struct addrinfo *result = NULL;
283  struct addrinfo hints = {0, AF_UNSPEC, SOCK_STREAM, IPPROTO_TCP, 0, NULL, NULL, NULL};
284 
285  if ((rc = getaddrinfo(addr, NULL, &hints, &result)) == 0) {
286  struct addrinfo* res = result;
287 
288  /* prefer ip4 addresses */
289  while (res) {
290  if (res->ai_family == AF_INET) {
291  result = res;
292  break;
293  }
294  res = res->ai_next;
295  }
296 
297  if (result->ai_family == AF_INET) {
298  address.sin_port = htons(port);
299  address.sin_family = family = AF_INET;
300  address.sin_addr = ((struct sockaddr_in*)(result->ai_addr))->sin_addr;
301  }
302  else {
303  rc = -1;
304  }
305 
306  freeaddrinfo(result);
307  }
308 
309  if (rc == 0) {
310  n->my_socket = socket(family, type, 0);
311  if (n->my_socket != -1)
312  rc = connect(n->my_socket, (struct sockaddr*)&address, sizeof(address));
313  }
314 
315  return rc;
316 }
