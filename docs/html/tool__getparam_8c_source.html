321 /* Split the argument of -E to 'certname' and 'passphrase' separated by colon.
322  * We allow ':' and '\' to be escaped by '\' so that we can use certificate
323  * nicknames containing ':'. See <https://sourceforge.net/p/curl/bugs/1196/>
324  * for details. */
325 #ifndef UNITTESTS
326 static
327 #endif
328 void parse_cert_parameter(const char *cert_parameter,
329  char **certname,
330  char **passphrase)
331 {
332  size_t param_length = strlen(cert_parameter);
333  size_t span;
334  const char *param_place = NULL;
335  char *certname_place = NULL;
336  *certname = NULL;
337  *passphrase = NULL;
338 
339  /* most trivial assumption: cert_parameter is empty */
340  if(param_length == 0)
341  return;
342 
343  /* next less trivial: cert_parameter starts 'pkcs11:' and thus
344  * looks like a RFC7512 PKCS#11 URI which can be used as-is.
345  * Also if cert_parameter contains no colon nor backslash, this
346  * means no passphrase was given and no characters escaped */
347  if(curl_strnequal(cert_parameter, "pkcs11:", 7) ||
348  !strpbrk(cert_parameter, ":\\")) {
349  *certname = strdup(cert_parameter);
350  return;
351  }
352  /* deal with escaped chars; find unescaped colon if it exists */
353  certname_place = malloc(param_length + 1);
354  if(!certname_place)
355  return;
356 
357  *certname = certname_place;
358  param_place = cert_parameter;
359  while(*param_place) {
360  span = strcspn(param_place, ":\\");
361  strncpy(certname_place, param_place, span);
362  param_place += span;
363  certname_place += span;
364  /* we just ate all the non-special chars. now we're on either a special
365  * char or the end of the string. */
366  switch(*param_place) {
367  case '\0':
368  break;
369  case '\\':
370  param_place++;
371  switch(*param_place) {
372  case '\0':
373  *certname_place++ = '\\';
374  break;
375  case '\\':
376  *certname_place++ = '\\';
377  param_place++;
378  break;
379  case ':':
380  *certname_place++ = ':';
381  param_place++;
382  break;
383  default:
384  *certname_place++ = '\\';
385  *certname_place++ = *param_place;
386  param_place++;
387  break;
388  }
389  break;
390  case ':':
391  /* Since we live in a world of weirdness and confusion, the win32
392  dudes can use : when using drive letters and thus c:\file:password
393  needs to work. In order not to break compatibility, we still use : as
394  separator, but we try to detect when it is used for a file name! On
395  windows. */
396 #ifdef WIN32
397  if(param_place &&
398  (param_place == &cert_parameter[1]) &&
399  (cert_parameter[2] == '\\' || cert_parameter[2] == '/') &&
400  (ISALPHA(cert_parameter[0])) ) {
401  /* colon in the second column, followed by a backslash, and the
402  first character is an alphabetic letter:
403 
404  this is a drive letter colon */
405  *certname_place++ = ':';
406  param_place++;
407  break;
408  }
409 #endif
410  /* escaped colons and Windows drive letter colons were handled
411  * above; if we're still here, this is a separating colon */
412  param_place++;
413  if(strlen(param_place) > 0) {
414  *passphrase = strdup(param_place);
415  }
416  goto done;
417  }
418  }
419 done:
420  *certname_place = '\0';
421 }

437 /* Get a size parameter for '--limit-rate' or '--max-filesize'.
438  * We support a 'G', 'M' or 'K' suffix too.
439  */
440 static ParameterError GetSizeParameter(struct GlobalConfig *global,
441  const char *arg,
442  const char *which,
443  curl_off_t *value_out)
444 {
445  char *unit;
446  curl_off_t value;
447 
448  if(curlx_strtoofft(arg, &unit, 0, &value)) {
449  warnf(global, "invalid number specified for %s\n", which);
450  return PARAM_BAD_USE;
451  }
452 
453  if(!*unit)
454  unit = (char *)"b";
455  else if(strlen(unit) > 1)
456  unit = (char *)"w"; /* unsupported */
457 
458  switch(*unit) {
459  case 'G':
460  case 'g':
461  if(value > (CURL_OFF_T_MAX / (1024*1024*1024)))
462  return PARAM_NUMBER_TOO_LARGE;
463  value *= 1024*1024*1024;
464  break;
465  case 'M':
466  case 'm':
467  if(value > (CURL_OFF_T_MAX / (1024*1024)))
468  return PARAM_NUMBER_TOO_LARGE;
469  value *= 1024*1024;
470  break;
471  case 'K':
472  case 'k':
473  if(value > (CURL_OFF_T_MAX / 1024))
474  return PARAM_NUMBER_TOO_LARGE;
475  value *= 1024;
476  break;
477  case 'b':
478  case 'B':
479  /* for plain bytes, leave as-is */
480  break;
481  default:
482  warnf(global, "unsupported %s unit. Use G, M, K or B!\n", which);
483  return PARAM_BAD_USE;
484  }
485  *value_out = value;
486  return PARAM_OK;
487 }

489 ParameterError getparameter(const char *flag, /* f or -long-flag */
490  char *nextarg, /* NULL if unset */
491  bool *usedarg, /* set to TRUE if the arg
492  has been used */
493  struct GlobalConfig *global,
494  struct OperationConfig *config)
495 {
496  char letter;
497  char subletter = '\0'; /* subletters can only occur on long options */
498  int rc;
499  const char *parse = NULL;
500  unsigned int j;
501  time_t now;
502  int hit = -1;
503  bool longopt = FALSE;
504  bool singleopt = FALSE; /* when true means '-o foo' used '-ofoo' */
505  ParameterError err;
506  bool toggle = TRUE; /* how to switch boolean options, on or off. Controlled
507  by using --OPTION or --no-OPTION */
508 
509  *usedarg = FALSE; /* default is that we don't use the arg */
510 
511  if(('-' != flag[0]) || ('-' == flag[1])) {
512  /* this should be a long name */
513  const char *word = ('-' == flag[0]) ? flag + 2 : flag;
514  size_t fnam = strlen(word);
515  int numhits = 0;
516 
517  if(!strncmp(word, "no-", 3)) {
518  /* disable this option but ignore the "no-" part when looking for it */
519  word += 3;
520  toggle = FALSE;
521  }
522 
523  for(j = 0; j < sizeof(aliases)/sizeof(aliases[0]); j++) {
524  if(curl_strnequal(aliases[j].lname, word, fnam)) {
525  longopt = TRUE;
526  numhits++;
527  if(curl_strequal(aliases[j].lname, word)) {
528  parse = aliases[j].letter;
529  hit = j;
530  numhits = 1; /* a single unique hit */
531  break;
532  }
533  parse = aliases[j].letter;
534  hit = j;
535  }
536  }
537  if(numhits > 1) {
538  /* this is at least the second match! */
539  return PARAM_OPTION_AMBIGUOUS;
540  }
541  if(hit < 0) {
542  return PARAM_OPTION_UNKNOWN;
543  }
544  }
545  else {
546  flag++; /* prefixed with one dash, pass it */
547  hit = -1;
548  parse = flag;
549  }
550 
551  do {
552  /* we can loop here if we have multiple single-letters */
553 
554  if(!longopt) {
555  letter = (char)*parse;
556  subletter = '\0';
557  }
558  else {
559  letter = parse[0];
560  subletter = parse[1];
561  }
562 
563  if(hit < 0) {
564  for(j = 0; j < sizeof(aliases)/sizeof(aliases[0]); j++) {
565  if(letter == aliases[j].letter[0]) {
566  hit = j;
567  break;
568  }
569  }
570  if(hit < 0) {
