231 void BLEAdvertisedDevice::parseAdvertisement(uint8_t* payload, size_t total_len) {
232  uint8_t length;
233  uint8_t ad_type;
234  uint8_t sizeConsumed = 0;
235  bool finished = false;
236  m_payload = payload;
237  m_payloadLength = total_len;
238 
239  while(!finished) {
240  length = *payload; // Retrieve the length of the record.
241  payload++; // Skip to type
242  sizeConsumed += 1 + length; // increase the size consumed.
243 
244  if (length != 0) { // A length of 0 indicates that we have reached the end.
245  ad_type = *payload;
246  payload++;
247  length--;
248 
249  char* pHex = BLEUtils::buildHexData(nullptr, payload, length);
250  ESP_LOGD(LOG_TAG, "Type: 0x%.2x (%s), length: %d, data: %s",
251  ad_type, BLEUtils::advTypeToString(ad_type), length, pHex);
252  free(pHex);
253 
254  switch(ad_type) {
255  case ESP_BLE_AD_TYPE_NAME_CMPL: { // Adv Data Type: 0x09
256  setName(std::string(reinterpret_cast<char*>(payload), length));
257  break;
258  } // ESP_BLE_AD_TYPE_NAME_CMPL
259 
260  case ESP_BLE_AD_TYPE_TX_PWR: { // Adv Data Type: 0x0A
261  setTXPower(*payload);
262  break;
263  } // ESP_BLE_AD_TYPE_TX_PWR
264 
265  case ESP_BLE_AD_TYPE_APPEARANCE: { // Adv Data Type: 0x19
266  setAppearance(*reinterpret_cast<uint16_t*>(payload));
267  break;
268  } // ESP_BLE_AD_TYPE_APPEARANCE
269 
270  case ESP_BLE_AD_TYPE_FLAG: { // Adv Data Type: 0x01
271  setAdFlag(*payload);
272  break;
273  } // ESP_BLE_AD_TYPE_FLAG
274 
275  case ESP_BLE_AD_TYPE_16SRV_CMPL:
276  case ESP_BLE_AD_TYPE_16SRV_PART: { // Adv Data Type: 0x02
277  for (int var = 0; var < length/2; ++var) {
278  setServiceUUID(BLEUUID(*reinterpret_cast<uint16_t*>(payload + var * 2)));
279  }
280  break;
281  } // ESP_BLE_AD_TYPE_16SRV_PART
282 
283  case ESP_BLE_AD_TYPE_32SRV_CMPL:
284  case ESP_BLE_AD_TYPE_32SRV_PART: { // Adv Data Type: 0x04
285  for (int var = 0; var < length/4; ++var) {
286  setServiceUUID(BLEUUID(*reinterpret_cast<uint32_t*>(payload + var * 4)));
287  }
288  break;
289  } // ESP_BLE_AD_TYPE_32SRV_PART
290 
291  case ESP_BLE_AD_TYPE_128SRV_CMPL: { // Adv Data Type: 0x07
292  setServiceUUID(BLEUUID(payload, 16, false));
293  break;
294  } // ESP_BLE_AD_TYPE_128SRV_CMPL
295 
296  case ESP_BLE_AD_TYPE_128SRV_PART: { // Adv Data Type: 0x06
297  setServiceUUID(BLEUUID(payload, 16, false));
298  break;
299  } // ESP_BLE_AD_TYPE_128SRV_PART
300 
301  // See CSS Part A 1.4 Manufacturer Specific Data
302  case ESP_BLE_AD_MANUFACTURER_SPECIFIC_TYPE: {
303  setManufacturerData(std::string(reinterpret_cast<char*>(payload), length));
304  break;
305  } // ESP_BLE_AD_MANUFACTURER_SPECIFIC_TYPE
306 
307  case ESP_BLE_AD_TYPE_SERVICE_DATA: { // Adv Data Type: 0x16 (Service Data) - 2 byte UUID
308  if (length < 2) {
309  ESP_LOGE(LOG_TAG, "Length too small for ESP_BLE_AD_TYPE_SERVICE_DATA");
310  break;
311  }
312  uint16_t uuid = *(uint16_t*)payload;
313  setServiceDataUUID(BLEUUID(uuid));
314  if (length > 2) {
315  setServiceData(std::string(reinterpret_cast<char*>(payload + 2), length - 2));
316  }
317  break;
318  } //ESP_BLE_AD_TYPE_SERVICE_DATA
319 
320  case ESP_BLE_AD_TYPE_32SERVICE_DATA: { // Adv Data Type: 0x20 (Service Data) - 4 byte UUID
321  if (length < 4) {
322  ESP_LOGE(LOG_TAG, "Length too small for ESP_BLE_AD_TYPE_32SERVICE_DATA");
323  break;
324  }
325  uint32_t uuid = *(uint32_t*) payload;
326  setServiceDataUUID(BLEUUID(uuid));
327  if (length > 4) {
328  setServiceData(std::string(reinterpret_cast<char*>(payload + 4), length - 4));
329  }
330  break;
331  } //ESP_BLE_AD_TYPE_32SERVICE_DATA
332 
333  case ESP_BLE_AD_TYPE_128SERVICE_DATA: { // Adv Data Type: 0x21 (Service Data) - 16 byte UUID
334  if (length < 16) {
335  ESP_LOGE(LOG_TAG, "Length too small for ESP_BLE_AD_TYPE_128SERVICE_DATA");
336  break;
337  }
338 
339  setServiceDataUUID(BLEUUID(payload, (size_t)16, false));
340  if (length > 16) {
341  setServiceData(std::string(reinterpret_cast<char*>(payload + 16), length - 16));
342  }
343  break;
344  } //ESP_BLE_AD_TYPE_32SERVICE_DATA
345 
346  default: {
347  ESP_LOGD(LOG_TAG, "Unhandled type: adType: %d - 0x%.2x", ad_type, ad_type);
348  break;
349  }
350  } // switch
351  payload += length;
352  } // Length <> 0
353 
354 
355  if (sizeConsumed >= total_len)
356  finished = true;
357 
358  } // !finished
359 } // parseAdvertisement

492  ss << "Name: " << getName() << ", Address: " << getAddress().toString();
493  if (haveAppearance()) {

499  free(pHex);
