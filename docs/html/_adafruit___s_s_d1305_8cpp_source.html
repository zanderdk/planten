136 void Adafruit_SSD1305::drawPixel(int16_t x, int16_t y, uint16_t color) {
137  if ((x >= width()) || (y >= height()) || (x < 0) || (y < 0)) {
138  return;
139  }
140 
141  // check rotation, move pixel around if necessary
142  switch (getRotation()) {
143  case 1:
144  adagfx_swap(x, y);
145  x = WIDTH - x - 1;
146  break;
147  case 2:
148  x = WIDTH - x - 1;
149  y = HEIGHT - y - 1;
150  break;
151  case 3:
152  adagfx_swap(x, y);
153  y = HEIGHT - y - 1;
154  break;
155  }
156 
157  // x is which column
158  if (color == WHITE) {
159  buffer[x+ (y/8)*SSD1305_LCDWIDTH] |= _BV((y%8));
160  }
161  else {
162  buffer[x+ (y/8)*SSD1305_LCDWIDTH] &= ~_BV((y%8));
163  }
164 }
