74 static CURLcode gopher_do(struct connectdata *conn, bool *done)
75 {
76  CURLcode result = CURLE_OK;
77  struct Curl_easy *data = conn->data;
78  curl_socket_t sockfd = conn->sock[FIRSTSOCKET];
79 
80  curl_off_t *bytecount = &data->req.bytecount;
81  char *path = data->state.up.path;
82  char *sel = NULL;
83  char *sel_org = NULL;
84  ssize_t amount, k;
85  size_t len;
86 
87  *done = TRUE; /* unconditionally */
88 
89  /* Create selector. Degenerate cases: / and /1 => convert to "" */
90  if(strlen(path) <= 2) {
91  sel = (char *)"";
92  len = strlen(sel);
93  }
94  else {
95  char *newp;
96 
97  /* Otherwise, drop / and the first character (i.e., item type) ... */
98  newp = path;
99  newp += 2;
100 
101  /* ... and finally unescape */
102  result = Curl_urldecode(data, newp, 0, &sel, &len, FALSE);
103  if(result)
104  return result;
105  sel_org = sel;
106  }
107 
108  /* We use Curl_write instead of Curl_sendf to make sure the entire buffer is
109  sent, which could be sizeable with long selectors. */
110  k = curlx_uztosz(len);
111 
112  for(;;) {
113  result = Curl_write(conn, sockfd, sel, k, &amount);
114  if(!result) { /* Which may not have written it all! */
115  result = Curl_client_write(conn, CLIENTWRITE_HEADER, sel, amount);
116  if(result)
117  break;
118 
119  k -= amount;
120  sel += amount;
121  if(k < 1)
122  break; /* but it did write it all */
123  }
124  else
125  break;
126 
127  /* Don't busyloop. The entire loop thing is a work-around as it causes a
128  BLOCKING behavior which is a NO-NO. This function should rather be
129  split up in a do and a doing piece where the pieces that aren't
130  possible to send now will be sent in the doing function repeatedly
131  until the entire request is sent.
132 
133  Wait a while for the socket to be writable. Note that this doesn't
134  acknowledge the timeout.
135  */
136  if(SOCKET_WRITABLE(sockfd, 100) < 0) {
137  result = CURLE_SEND_ERROR;
138  break;
139  }
140  }
141 
142  free(sel_org);
143 
144  if(!result)
145  /* We can use Curl_sendf to send the terminal \r\n relatively safely and
146  save allocing another string/doing another _write loop. */
147  result = Curl_sendf(sockfd, conn, "\r\n");
148  if(result) {
149  failf(data, "Failed sending Gopher request");
150  return result;
151  }
152  result = Curl_client_write(conn, CLIENTWRITE_HEADER, (char *)"\r\n", 2);
153  if(result)
154  return result;
155 
156  Curl_setup_transfer(conn, FIRSTSOCKET, -1, FALSE, bytecount,
157  -1, NULL); /* no upload */
158  return CURLE_OK;
159 }
