23  * multi socket API usage with libevent 2
24  * </DESC>

26 /* Example application source code using the multi socket interface to
27  download many files at once.
28 
29 Written by Jeff Pohlmeyer
30 
31 Requires libevent version 2 and a (POSIX?) system that has mkfifo().
32 
33 This is an adaptation of libcurl's "hipev.c" and libevent's "event-test.c"
34 sample programs.
35 
36 When running, the program creates the named pipe "hiper.fifo"
37 
38 Whenever there is input into the fifo, the program reads the input as a list
39 of URL's and creates some new easy handles to fetch each URL via the
40 curl_multi "hiper" API.
41 
42 
43 Thus, you can try a single URL:
44  % echo http://www.yahoo.com > hiper.fifo
45 
46 Or a whole bunch of them:
47  % cat my-url-list > hiper.fifo
48 
49 The fifo buffer is handled almost instantly, so you can even add more URL's
50 while the previous requests are still being downloaded.
51 
52 Note:
53  For the sake of simplicity, URL length is limited to 1023 char's !
54 
55 This is purely a demo app, all retrieved data is simply discarded by the write
56 callback.
57 
58 */

132  __case(CURLM_LAST); break;

145 /* Update the event timer after curl_multi library calls */
146 static int multi_timer_cb(CURLM *multi _Unused, long timeout_ms, GlobalInfo *g)
147 {
148  struct timeval timeout;
149  CURLMcode rc;
150 
151  timeout.tv_sec = timeout_ms/1000;
152  timeout.tv_usec = (timeout_ms%1000)*1000;
153  fprintf(MSG_OUT, "multi_timer_cb: Setting timeout to %ld ms\n", timeout_ms);
154 
155  /* TODO
156  *
157  * if timeout_ms is 0, call curl_multi_socket_action() at once!
158  *
159  * if timeout_ms is -1, just delete the timer
160  *
161  * for all other values of timeout_ms, this should set or *update*
162  * the timer to the new value
163  */
164  if(timeout_ms == 0) {
165  rc = curl_multi_socket_action(g->multi,
166  CURL_SOCKET_TIMEOUT, 0, &g->still_running);
167  mcode_or_die("multi_timer_cb: curl_multi_socket_action", rc);
168  }
169  else if(timeout_ms == -1)
170  evtimer_del(&g->timer_event);
171  else
172  evtimer_add(&g->timer_event, &timeout);
173  return 0;
174 }

177 /* Check for completed transfers, and remove their easy handles */
178 static void check_multi_info(GlobalInfo *g)
179 {
180  char *eff_url;
181  CURLMsg *msg;
182  int msgs_left;
183  ConnInfo *conn;
184  CURL *easy;
185  CURLcode res;
186 
187  fprintf(MSG_OUT, "REMAINING: %d\n", g->still_running);
188  while((msg = curl_multi_info_read(g->multi, &msgs_left))) {
189  if(msg->msg == CURLMSG_DONE) {
190  easy = msg->easy_handle;
191  res = msg->data.result;
192  curl_easy_getinfo(easy, CURLINFO_PRIVATE, &conn);
193  curl_easy_getinfo(easy, CURLINFO_EFFECTIVE_URL, &eff_url);
194  fprintf(MSG_OUT, "DONE: %s => (%d) %s\n", eff_url, res, conn->error);
195  curl_multi_remove_handle(g->multi, easy);
196  free(conn->url);
197  curl_easy_cleanup(easy);
198  free(conn);
199  }
200  }
201  if(g->still_running == 0 && g->stopped)
202  event_base_loopbreak(g->evbase);
203 }

285 {
286  GlobalInfo *g = (GlobalInfo*) cbp;

370 /* This gets called whenever data is received from the fifo */
371 static void fifo_cb(int fd _Unused, short event _Unused, void *arg)
372 {
373  char s[1024];
374  long int rv = 0;
375  int n = 0;
376  GlobalInfo *g = (GlobalInfo *)arg;
377 
378  do {
379  s[0]='\0';
380  rv = fscanf(g->input, "%1023s%n", s, &n);
381  s[n]='\0';
382  if(n && s[0]) {
383  if(!strcmp(s, "stop")) {
384  g->stopped = 1;
385  if(g->still_running == 0)
386  event_base_loopbreak(g->evbase);
387  }
388  else
389  new_conn(s, arg); /* if we read a URL, go get it! */
390  }
391  else
392  break;
393  } while(rv != EOF);
394 }

396 /* Create a named pipe and tell libevent to monitor it */
397 static const char *fifo = "hiper.fifo";
398 static int init_fifo(GlobalInfo *g)
399 {
400  struct stat st;
401  curl_socket_t sockfd;
402 
403  fprintf(MSG_OUT, "Creating named pipe \"%s\"\n", fifo);
404  if(lstat (fifo, &st) == 0) {
405  if((st.st_mode & S_IFMT) == S_IFREG) {
406  errno = EEXIST;
407  perror("lstat");
408  exit(1);
409  }
410  }
411  unlink(fifo);
412  if(mkfifo (fifo, 0600) == -1) {
413  perror("mkfifo");
414  exit(1);
415  }
416  sockfd = open(fifo, O_RDWR | O_NONBLOCK, 0);
417  if(sockfd == -1) {
418  perror("open");
419  exit(1);
420  }
421  g->input = fdopen(sockfd, "r");
422 
423  fprintf(MSG_OUT, "Now, pipe some URL's into > %s\n", fifo);
424  event_assign(&g->fifo_event, g->evbase, sockfd, EV_READ|EV_PERSIST,
425  fifo_cb, g);
426  event_add(&g->fifo_event, NULL);
427  return (0);
428 }
