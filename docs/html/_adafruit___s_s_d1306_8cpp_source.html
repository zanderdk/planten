112 // the most basic function, set a single pixel
113 void Adafruit_SSD1306::drawPixel(int16_t x, int16_t y, uint16_t color) {
114  if ((x < 0) || (x >= width()) || (y < 0) || (y >= height())) {
115  return;
116  }
117 
118  // check rotation, move pixel around if necessary
119  switch (getRotation()) {
120  case 1:
121  ssd1306_swap(x, y);
122  x = WIDTH - x - 1;
123  break;
124  case 2:
125  x = WIDTH - x - 1;
126  y = HEIGHT - y - 1;
127  break;
128  case 3:
129  ssd1306_swap(x, y);
130  y = HEIGHT - y - 1;
131  break;
132  }
133 
134  // x is which column
135  switch (color)
136  {
137  case WHITE: buffer[x+ (y/8)*SSD1306_LCDWIDTH] |= (1 << (y&7)); break;
138  case BLACK: buffer[x+ (y/8)*SSD1306_LCDWIDTH] &= ~(1 << (y&7)); break;
139  case INVERSE: buffer[x+ (y/8)*SSD1306_LCDWIDTH] ^= (1 << (y&7)); break;
140  }
141 }

478 void Adafruit_SSD1306::drawFastHLineInternal(int16_t x, int16_t y, int16_t w, uint16_t color) {
479  // Do bounds/limit checks
480  if(y < 0 || y >= HEIGHT) { return; }
481 
482  // make sure we don't try to draw below 0
483  if(x < 0) {
484  w += x;
485  x = 0;
486  }
487 
488  // make sure we don't go off the edge of the display
489  if( (x + w) > WIDTH) {
490  w = (WIDTH - x);
491  }
492 
493  // if our width is now negative, punt
494  if(w <= 0) { return; }
495 
496  // set up the pointer for movement through the buffer
497  register uint8_t *pBuf = buffer;
498  // adjust the buffer pointer for the current row
499  pBuf += ((y/8) * SSD1306_LCDWIDTH);
500  // and offset x columns in
501  pBuf += x;
502 
503  register uint8_t mask = 1 << (y&7);
504 
505  switch (color)
506  {
507  case WHITE: while(w--) { *pBuf++ |= mask; }; break;
508  case BLACK: mask = ~mask; while(w--) { *pBuf++ &= mask; }; break;
509  case INVERSE: while(w--) { *pBuf++ ^= mask; }; break;
510  }

547 void Adafruit_SSD1306::drawFastVLineInternal(int16_t x, int16_t __y, int16_t __h, uint16_t color) {
548 
549  // do nothing if we're off the left or right side of the screen
550  if(x < 0 || x >= WIDTH) { return; }
551 
552  // make sure we don't try to draw below 0
553  if(__y < 0) {
554  // __y is negative, this will subtract enough from __h to account for __y being 0
555  __h += __y;
556  __y = 0;
557 
558  }
559 
560  // make sure we don't go past the height of the display
561  if( (__y + __h) > HEIGHT) {
562  __h = (HEIGHT - __y);
563  }
564 
565  // if our height is now negative, punt
566  if(__h <= 0) {
567  return;
568  }
569 
570  // this display doesn't need ints for coordinates, use local byte registers for faster juggling
571  register uint8_t y = __y;
572  register uint8_t h = __h;
573 
574 
575  // set up the pointer for fast movement through the buffer
576  register uint8_t *pBuf = buffer;
577  // adjust the buffer pointer for the current row
578  pBuf += ((y/8) * SSD1306_LCDWIDTH);
579  // and offset x columns in
580  pBuf += x;
581 
582  // do the first partial byte, if necessary - this requires some masking
583  register uint8_t mod = (y&7);
584  if(mod) {
585  // mask off the high n bits we want to set
586  mod = 8-mod;
587 
588  // note - lookup table results in a nearly 10% performance improvement in fill* functions
589  // register uint8_t mask = ~(0xFF >> (mod));
590  static uint8_t premask[8] = {0x00, 0x80, 0xC0, 0xE0, 0xF0, 0xF8, 0xFC, 0xFE };
591  register uint8_t mask = premask[mod];
592 
593  // adjust the mask if we're not going to reach the end of this byte
594  if( h < mod) {
595  mask &= (0XFF >> (mod-h));
596  }
597 
598  switch (color)
599  {
600  case WHITE: *pBuf |= mask; break;
601  case BLACK: *pBuf &= ~mask; break;
602  case INVERSE: *pBuf ^= mask; break;
603  }
604 
605  // fast exit if we're done here!
606  if(h<mod) { return; }
607 
608  h -= mod;
609 
610  pBuf += SSD1306_LCDWIDTH;
611  }
612 
613 
614  // write solid bytes while we can - effectively doing 8 rows at a time
615  if(h >= 8) {
616  if (color == INVERSE) { // separate copy of the code so we don't impact performance of the black/white write version with an extra comparison per loop
617  do {
618  *pBuf=~(*pBuf);
619 
620  // adjust the buffer forward 8 rows worth of data
621  pBuf += SSD1306_LCDWIDTH;
622 
623  // adjust h & y (there's got to be a faster way for me to do this, but this should still help a fair bit for now)
624  h -= 8;
625  } while(h >= 8);
626  }
627  else {
628  // store a local value to work with
629  register uint8_t val = (color == WHITE) ? 255 : 0;
630 
631  do {
632  // write our value in
633  *pBuf = val;
634 
635  // adjust the buffer forward 8 rows worth of data
636  pBuf += SSD1306_LCDWIDTH;
637 
638  // adjust h & y (there's got to be a faster way for me to do this, but this should still help a fair bit for now)
639  h -= 8;
640  } while(h >= 8);
641  }
642  }
643 
644  // now do the final partial byte, if necessary
645  if(h) {
646  mod = h & 7;
647  // this time we want to mask the low bits of the byte, vs the high bits we did above
648  // register uint8_t mask = (1 << mod) - 1;
649  // note - lookup table results in a nearly 10% performance improvement in fill* functions
650  static uint8_t postmask[8] = {0x00, 0x01, 0x03, 0x07, 0x0F, 0x1F, 0x3F, 0x7F };
651  register uint8_t mask = postmask[mod];
652  switch (color)
653  {
654  case WHITE: *pBuf |= mask; break;
655  case BLACK: *pBuf &= ~mask; break;
656  case INVERSE: *pBuf ^= mask; break;
657  }
658  }
