68 int thread_setup(void)
69 {

72  mutex_buf = malloc(CRYPTO_num_locks() * sizeof(MUTEX_TYPE));
73  if(!mutex_buf)
74  return 0;
75  for(i = 0; i < CRYPTO_num_locks(); i++)
76  MUTEX_SETUP(mutex_buf[i]);
