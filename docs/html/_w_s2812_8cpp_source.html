151 void WS2812::show() {
152  auto pCurrentItem = this->items;
153 
154  for (uint16_t i = 0; i < this->pixelCount; i++) {
155  uint32_t currentPixel =
156  (getChannelValueByType(this->colorOrder[0], this->pixels[i]) << 16) |
157  (getChannelValueByType(this->colorOrder[1], this->pixels[i]) << 8) |
158  (getChannelValueByType(this->colorOrder[2], this->pixels[i]));
159 
160  ESP_LOGD(LOG_TAG, "Pixel value: %x", currentPixel);
161  for (int8_t j = 23; j >= 0; j--) {
162  // We have 24 bits of data representing the red, green amd blue channels. The value of the
163  // 24 bits to output is in the variable current_pixel. We now need to stream this value
164  // through RMT in most significant bit first. To do this, we iterate through each of the 24
165  // bits from MSB to LSB.
166  if (currentPixel & (1 << j)) {
167  setItem1(pCurrentItem);
168  } else {
169  setItem0(pCurrentItem);
170  }
171  pCurrentItem++;
172  }
173  }
174  setTerminator(pCurrentItem); // Write the RMT terminator.
175 
176  // Show the pixels.
177  ESP_ERROR_CHECK(rmt_write_items(this->channel, this->items, this->pixelCount * 24, 1 /* wait till done */));
178 } // show
