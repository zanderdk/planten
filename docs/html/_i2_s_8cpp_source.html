95 uint32_t DMABuffer::getData(uint8_t* pData, uint32_t length) {
96  uint8_t* pBuf = (uint8_t*) m_desc.buf;
97  if (length > getLength()) length = getLength();
98  uint32_t i;
99  // The descriptor buffer is filled with data that contains:
100  // b1 00 b0 00 b3 00 b2 00 b5 00 b4 00 b7 00 b6 00
101  // Our goal is to populate the passed in buffer with data of the form:
102  // b0 b1 b2 b3 b4 b5 b6 b7 ...
103  // The following alogrithm does that.
104  for (i = 0; i < length; i += 2) {
105  *pData = pBuf[2];
106  pData++;
107  *pData = pBuf[0];
108  pData++;
109  pBuf += 4;
110  }
111  return i;
112 } // getData
