42 static void memrcpy(uint8_t* target, uint8_t* source, uint32_t size) {
43  assert(size > 0);
44  target += (size - 1); // Point target to the last byte of the target data

71 BLEUUID::BLEUUID(std::string value) {
72  m_valueSet = true;
73  if (value.length() == 4) {
74  m_uuid.len = ESP_UUID_LEN_16;
75  m_uuid.uuid.uuid16 = 0;
76  for(int i=0;i<value.length();){
77  uint8_t MSB = value.c_str()[i];
78  uint8_t LSB = value.c_str()[i+1];
79 
80  if(MSB > '9') MSB -= 7;
81  if(LSB > '9') LSB -= 7;
82  m_uuid.uuid.uuid16 += (((MSB&0x0F) <<4) | (LSB & 0x0F))<<(2-i)*4;
83  i+=2;
84  }
85  }
86  else if (value.length() == 8) {
87  m_uuid.len = ESP_UUID_LEN_32;
88  m_uuid.uuid.uuid32 = 0;
89  for(int i=0;i<value.length();){
90  uint8_t MSB = value.c_str()[i];
91  uint8_t LSB = value.c_str()[i+1];
92 
93  if(MSB > '9') MSB -= 7;
94  if(LSB > '9') LSB -= 7;
95  m_uuid.uuid.uuid32 += (((MSB&0x0F) <<4) | (LSB & 0x0F))<<(6-i)*4;
96  i+=2;
97  }
98  }
99  else if (value.length() == 16) { // how we can have 16 byte length string reprezenting 128 bit uuid??? needs to be investigated (lack of time)
100  m_uuid.len = ESP_UUID_LEN_128;
101  memrcpy(m_uuid.uuid.uuid128, (uint8_t*)value.data(), 16);
102  }
103  else if (value.length() == 36) {
104  // If the length of the string is 36 bytes then we will assume it is a long hex string in
105  // UUID format.
106  m_uuid.len = ESP_UUID_LEN_128;
107  int n = 0;
108  for(int i=0;i<value.length();){
109  if(value.c_str()[i] == '-')
110  i++;
111  uint8_t MSB = value.c_str()[i];
112  uint8_t LSB = value.c_str()[i+1];
113 
114  if(MSB > '9') MSB -= 7;
115  if(LSB > '9') LSB -= 7;
116  m_uuid.uuid.uuid128[15-n++] = ((MSB&0x0F) <<4) | (LSB & 0x0F);
117  i+=2;
118  }
119  }
120  else {
121  ESP_LOGE(LOG_TAG, "ERROR: UUID value not 2, 4, 16 or 36 bytes");
122  m_valueSet = false;
123  }
124 } //BLEUUID(std::string)

253 BLEUUID BLEUUID::fromString(std::string _uuid) {

255  if (strstr(_uuid.c_str(), "0x") != nullptr) { // If the string starts with 0x, skip those characters.
256  start = 2;

259 
260  if(len == 4) {
261  uint16_t x = strtoul(_uuid.substr(start, len).c_str(), NULL, 16);
262  return BLEUUID(x);
263  } else if (len == 8) {
264  uint32_t x = strtoul(_uuid.substr(start, len).c_str(), NULL, 16);
265  return BLEUUID(x);

267  return BLEUUID(_uuid);
