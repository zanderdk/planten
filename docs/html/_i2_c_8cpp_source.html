97 void I2C::init(uint8_t address, gpio_num_t sdaPin, gpio_num_t sclPin, uint32_t clockSpeed, i2c_port_t portNum, bool pullup) {
98  ESP_LOGD(LOG_TAG, ">> I2c::init. address=%d, sda=%d, scl=%d, clockSpeed=%d, portNum=%d", address, sdaPin, sclPin, clockSpeed, portNum);
99  assert(portNum < I2C_NUM_MAX);
100  m_portNum = portNum;
101  m_sdaPin = sdaPin;
102  m_sclPin = sclPin;
103  m_address = address;
104 
105  i2c_config_t conf;
106  conf.mode = I2C_MODE_MASTER;
107  conf.sda_io_num = sdaPin;
108  conf.scl_io_num = sclPin;
109  conf.sda_pullup_en = pullup ? GPIO_PULLUP_ENABLE : GPIO_PULLUP_DISABLE;
110  conf.scl_pullup_en = pullup ? GPIO_PULLUP_ENABLE : GPIO_PULLUP_DISABLE;
111  conf.master.clk_speed = clockSpeed;
112  esp_err_t errRc = ::i2c_param_config(m_portNum, &conf);
113  if (errRc != ESP_OK) {
114  ESP_LOGE(LOG_TAG, "i2c_param_config: rc=%d %s", errRc, GeneralUtils::errorToString(errRc));
115  }
116  if (!driverInstalled) {
117  errRc = ::i2c_driver_install(m_portNum, I2C_MODE_MASTER, 0, 0, 0);
118  if (errRc != ESP_OK) {
119  ESP_LOGE(LOG_TAG, "i2c_driver_install: rc=%d %s", errRc, GeneralUtils::errorToString(errRc));
120  }
121  driverInstalled = true;
122  }
123 } // init

224 bool I2C::slavePresent(uint8_t address) {
225  i2c_cmd_handle_t cmd = ::i2c_cmd_link_create();
226  ::i2c_master_start(cmd);
227  ::i2c_master_write_byte(cmd, (address << 1) | I2C_MASTER_WRITE, 1 /* expect ack */);
228  ::i2c_master_stop(cmd);
229 
230  esp_err_t espRc = ::i2c_master_cmd_begin(m_portNum, cmd, 100 / portTICK_PERIOD_MS);
231  i2c_cmd_link_delete(cmd);
232  return espRc == 0; // Return true if the slave is present and false otherwise.
233 } // slavePresent
